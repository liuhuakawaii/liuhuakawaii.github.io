JavaScriptз”ҹжҲҗйҡҸжңәйўңиүІзҡ„еӨҡз§Қж–№жі• рҹҺЁ вҖӢ

еңЁеүҚз«ҜејҖеҸ‘дёӯпјҢжңүж—¶жҲ‘д»¬йңҖиҰҒз”ҹжҲҗйҡҸжңәйўңиүІжқҘеўһеҠ йЎөйқўзҡ„и¶Је‘іжҖ§жҲ–иҖ…з”ЁдәҺж•°жҚ®еҸҜи§ҶеҢ–гҖӮжң¬ж–Үе°Ҷд»Ӣз»ҚеҮ з§Қеёёз”Ёзҡ„з”ҹжҲҗйҡҸжңәйўңиүІзҡ„ж–№жі•гҖӮ

1. RGBж јејҸз”ҹжҲҗ рҹ”ў вҖӢ

жңҖеҹәзЎҖзҡ„ж–№ејҸжҳҜйҖҡиҝҮRGBдёүеҺҹиүІжқҘз”ҹжҲҗпјҡ

js
function randomColor() {
  const r = Math.floor(Math.random() * 256)  // 0-255
  const g = Math.floor(Math.random() * 256)  // 0-255
  const b = Math.floor(Math.random() * 256)  // 0-255
  return `rgb(${r},${g},${b})`
}

console.log(randomColor()) // иҫ“еҮәзұ»дјј: rgb(123,45,67)

2. еҚҒе…ӯиҝӣеҲ¶ж јејҸз”ҹжҲҗ #пёҸвғЈ вҖӢ

йҖҡиҝҮе°ҶйҡҸжңәж•°иҪ¬жҚўдёә16иҝӣеҲ¶жқҘз”ҹжҲҗйўңиүІд»Јз Ғпјҡ

js
function randomHexColor() {
  return '#' + Math.floor(Math.random() * 16777215).toString(16).padStart(6, '0')
}

console.log(randomHexColor()) // иҫ“еҮәзұ»дјј: #7B2D43

3. HSLж јејҸз”ҹжҲҗ рҹҢҲ вҖӢ

HSLж јејҸжӣҙе®№жҳ“жҺ§еҲ¶йўңиүІзҡ„йҘұе’ҢеәҰе’Ңдә®еәҰпјҡ

js
function randomHslColor() {
  const h = Math.floor(Math.random() * 360) // иүІзӣё 0-360
  const s = Math.floor(Math.random() * 100) // йҘұе’ҢеәҰ 0-100
  const l = Math.floor(Math.random() * 100) // дә®еәҰ 0-100
  return `hsl(${h},${s}%,${l}%)`
}

console.log(randomHslColor()) // иҫ“еҮәзұ»дјј: hsl(230,45%,60%)

жіЁж„ҸдәӢйЎ№ вҡ пёҸ вҖӢ

  1. дҪҝз”ЁMath.floor()жҲ–дҪҚиҝҗз®—з¬Ұ|0йғҪеҸҜд»Ҙе°Ҷе°Ҹж•°иҪ¬дёәж•ҙж•°пјҢдҪҶMath.floor()зҡ„еҸҜиҜ»жҖ§жӣҙеҘҪ
  2. еҚҒе…ӯиҝӣеҲ¶ж–№жі•дёӯдҪҝз”ЁpadStart(6, '0')зЎ®дҝқз”ҹжҲҗзҡ„йўңиүІд»Јз Ғе§Ӣз»ҲжҳҜ6дҪҚ
  3. еңЁе®һйҷ…еә”з”ЁдёӯпјҢеҸҜиғҪйңҖиҰҒжҺ§еҲ¶з”ҹжҲҗйўңиүІзҡ„иҢғеӣҙпјҢжҜ”еҰӮйҒҝе…ҚиҝҮдә®жҲ–иҝҮжҡ—зҡ„йўңиүІ

е®һз”ЁзӨәдҫӢ рҹ’Ў вҖӢ

з”ҹжҲҗжҹ”е’Ңзҡ„йўңиүІ вҖӢ

js
function randomPastelColor() {
  const h = Math.floor(Math.random() * 360)
  return `hsl(${h}, 70%, 80%)`  // еӣәе®ҡйҘұе’ҢеәҰе’Ңдә®еәҰпјҢз”ҹжҲҗжҹ”е’Ңзҡ„иүІи°ғ
}

з”ҹжҲҗзү№е®ҡиүІзі»зҡ„йҡҸжңәйўңиүІ вҖӢ

js
function randomColorInRange(hueStart, hueEnd) {
  const h = hueStart + Math.floor(Math.random() * (hueEnd - hueStart))
  const s = 70 + Math.floor(Math.random() * 30) // 70-100
  const l = 40 + Math.floor(Math.random() * 30) // 40-70
  return `hsl(${h},${s}%,${l}%)`
}

// з”ҹжҲҗйҡҸжңәи“қиүІ
console.log(randomColorInRange(200, 240))

жҖ§иғҪиҖғиҷ‘ рҹҡҖ вҖӢ

  • RGBе’ҢеҚҒе…ӯиҝӣеҲ¶ж–№жі•зҡ„и®Ўз®—йҮҸиҫғе°ҸпјҢйҖӮеҗҲеӨ§йҮҸйўңиүІз”ҹжҲҗзҡ„еңәжҷҜ
  • HSLж–№жі•иҷҪз„¶жҺ§еҲ¶жӣҙзІҫзЎ®пјҢдҪҶи®Ўз®—йҮҸзӣёеҜ№иҫғеӨ§
  • еҰӮжһңйңҖиҰҒеӨ§йҮҸз”ҹжҲҗйўңиүІпјҢе»әи®®е°Ҷз»“жһңзј“еӯҳдҪҝз”Ё

жөҸи§ҲеҷЁе…је®№жҖ§ рҹҢҗ вҖӢ

жүҖжңүзҺ°д»ЈжөҸи§ҲеҷЁйғҪе®Ңе…Ёж”ҜжҢҒиҝҷдәӣйўңиүІж јејҸпјҢдҪҶеңЁеӨ„зҗҶйҖҸжҳҺеәҰж—¶пјҢе»әи®®дҪҝз”Ёrgba()жҲ–hsla()жқҘзЎ®дҝқжңҖдҪіе…је®№жҖ§гҖӮ

еёҢжңӣиҝҷдәӣж–№жі•иғҪеё®еҠ©дҪ еңЁйЎ№зӣ®дёӯиҪ»жқҫе®һзҺ°йҡҸжңәйўңиүІзҡ„з”ҹжҲҗпјҒрҹҺү
