##  Vue2的响应式原理 -- 源码剖析   

> vue2的响应式原理一直是面试基本必考的，今天我们通过手写来剖析响应式原理的实现  

###  手写响应式原理  

1. vue首先把所有data中的属性和方法耦合到了一起  

  ```js
       //响应式的数据变化，数据变化了我可以监控到数据的变化
        //数据的取值 和 更改值我们要监控到
        const vm = new Vue({
          data() {
            return { //代理数据
              name: 'zf',
              age: 20,
              address: {
                content: 'DDD',
                num: 56
              },
              hobby: ['eat', 'slipp', {
                a: 1
              }]
            }
          }
        })


      //将所有的方法都耦合到一起
        function Vue(options){//options就是用户的选项
          this._init(options) //默认就调用了init
        }
        initMixin(Vue) //扩展了init方法  
  ```   
2. 将用户选项挂载到vm实例上  
  ```js   
      function initMixin(Vue){
        Vue.prototype._init = function(options){//用于初始化操作
          //vue  vm.$options 就是获取用户的配置
          //我们使用的vue的时候 $nextTick $data $attr.....
          const vm = this 
          vm.$options = options//将用户的选项挂载到实例上   
          //初始化状态
          initState(vm);
        } 
      }  
  ```   
3. 对属性进行劫持  
  ```js  
      function initState(vm){
        const opts = vm.$options //获取所有的选项
        if(opts.data){
          initData(vm)
        }
      }  

      function initData(vm){
        let data = vm.$options.data // 可能是函数或对象
        data = typeof data === 'function' ? data.call(vm) : data
        
          vm._data = data  //我将返回的对象放到了_data上
          //对数据进行劫持 vue2里采用了一个api defineProperty
          observe(data)
          //将vm._data用vm来代理就可以了
          for(let key in data){
            proxy(vm,'_data',key)
          }    
      } 
  ```  
4. 为了获取属性方便，进行数据代理  
  ```js  
      function proxy(vm,target,key){
        Object.defineProperty(vm,key,{
          get(){
            return vm[target][key]
          }
        })
      }  
  ```  
5. Observe观测者  进行递归，将所有对象属性加上了get和set   
  ```js  
      class Observer {
        constructor(data) {
          //Object.defineProperty只能劫持已经存在的属性（vue2里会为此单独写一些api $set $delete）
          Object.defineProperty(data, '__ob__', {
            value: this,
            enumerable: false //将__ob__变成不可枚举（循环时无法获取，防止死循环）
          })
          data.__ob__ = this //给数据加了一个标识
          if (Array.isArray(data)) {
            //我们可以重写数组的方法，7个变异方法是可以修改数组本身的
            data.__proto__ = newArrayProto //需要保留原数组原有的特性，并且重写部分方法
            this.observeArray(data) //如果数组中放的是对象，可以监控到对象的变化
          } else {
            this.walk(data)
          }
        }
        walk(data) { //循环对象 对属性依次劫持
          //"重新定义"属性  性能差
          Object.keys(data).forEach(key => defineReactive(data, key, data[key]))
        }
        observeArray(data) {
          data.forEach(item => {
            observe(item)
          })
        }
      }
      export function defineReactive(target, key, value) { //闭包  属性劫持
        observe(value) //对所有的对象都进行属性劫持
        Object.defineProperty(target, key, {
          get() { //取值的时候  会执行get
            return value
          },
          set(newValue) { //修改的时候 会执行set
            if (newValue === value) return
            value = newValue
          }
        })
      }
      export function observe(data) {
        //对这个对象进行劫持
        if (typeof data !== 'object' || data == null) {
          return //只对对象进行劫持
        }
        //如果一个对象被劫持过了，那就不需要再被劫持了(要判断一个对象是否被劫持过，可以增添一个实例，用实例来判断是否被劫持过)
        if (data.__ob__ instanceof Observer) {
          return data.__ob__ //说明这个对象被代理过了
        }
        return new Observer(data)
      }  
  ```  
6. 重写数组中的部分方法  
  ```js  
      let oldArrayProto = Array.prototype // 获取数组的原型
      let newArrayProto =  Object.create(oldArrayProto)
      let methods = [
        //找到所有的变异方法
        'push',
        'pop',
        'shift',
        'unshift',
        'reverse',
        'sort',
        'splice'
      ] // concat slice 不会改变原数组
      methods.forEach(methods=>{
        newArrayProto[method] = function(...args){//这里重写了数组的方法
        const result =  oldArrayProto[method].call(this,...args)//内部调用原来的方法，函数的劫持，切片编程
        //我们需要对新增的数据再次进行劫持
        let inserted
        let ob = this.__ob__
        switch(method){
          case 'push':
          case'unshift':
              inserted =args;
              break
          case 'splice':
              inserted = args.slice(2)
          default:
              break
        }
        console.log(inserted) //新增的内容
        if(inserted){
          //对新增的内容再次进行观测
          ob.observeArray(inserted)
        }
        return result
        }
      })  
  ```  
7. 总结 ：
    1. vue2中对每一个属性都进行"重新定义"属性，这也是相对性能差的原因，
    2. 数组如果数据量过大，那么将会有溢出栈的问题，所以对数组部分方法进行改造，
    3. `__ob__`既可以为重写数组方法中拿到`class`类中的`observeArray`方法，又可以给检测过的属性加上标识，防止重复监测的问题.  
    4. 将`__ob__`变成不可枚举（循环时无法获取，防止死循环）
