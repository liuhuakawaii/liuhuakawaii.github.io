问题：以下那段代码运行效率更高（隐藏类） ​

js
const obj1 = { a: 1 }
const obj2 = { a: 1 }
const obj3 = { a: 1 }
js
const obj1 = { a: 1 }
const obj2 = { b: 1 }
const obj3 = { c: 1 }
  • 上边效率更高，重用了
js
//测试代码
console.time('a')
for (let i = 0; i < 1000000; ++i) {
  const obj = {}
  obj.a = 1
}
console.timeEnd('a')

console.time('b')
for (let i = 0; i < 1000000; ++i) {
  const obj = {}
  obj[`${i}`] = i
}
console.timeEnd('b')
  • 比较主流的 JS 引擎是 v8，这里假设是跑在 chrome 或 node 上，用的是 v8 引擎。
  • V8 是一个 c++ 实现的 js 解析引擎，内部利用 隐藏类(Hidden Class)的方式来存放 JS 对象。
  • 隐藏类的特性是:多个属性顺序一致的 jS 对象，会重用同一个隐藏类，减少 new Class 的开销
  • 所以左边生成1个隐藏类，而右边生成3个隐藏类，因此左边代码性能更好。

优化建议

  1. 在构造函数中初始化所有属性，尽量避免"添加动态属性"的场景
  2. 始终按照同样属性顺序构建对象，使得隐藏类可以得到重用，
  3. 不要加载未初始化或已经删除的属性。

问题：如何判断 object 为空 ​

js
// 不完整的方法
Object.keys(obj).length === 0         // 只能检测普通属性
JSON.stringify(obj) === '{}'          // 不能检测 Symbol 属性，且会忽略值为 undefined 的属性
for (let key in obj) return false     // 不能遍历 Symbol 属性，且会遍历原型链上的属性

// 更完整的检测方法
Reflect.ownKeys(obj).length === 0     // 能检测所有自身属性（包括 Symbol）

// 特殊情况示例
const obj = {
  [Symbol('a')]: 1,
  b: undefined
}
  1. Reflect.ownKeys(obj)确实是最严谨的方法，因为：
    • 可以获取对象自身的所有属性键（包括 Symbol 属性）
    • 不会遍历原型链
    • 不会忽略值为 undefined 的属性
  2. 还可以考虑使用 Object.getOwnPropertyNames(obj).length + Object.getOwnPropertySymbols(obj).length === 0 作为替代方案，效果相同但可读性稍差
  3. 如果需要考虑原型链上的属性，可以这样写：
js
function isCompletelyEmpty(obj) {
  return Reflect.ownKeys(obj).length === 0 && Object.getPrototypeOf(obj) === Object.prototype;
}

问题：JavaScript 的数据类型有哪些 ​

  • 基本数据类型：
    • Number（数字）：表示数值，包括整数和浮点数。
    • String（字符串）：表示文本数据，可以是任意字符序列。
    • Boolean（布尔值）：表示逻辑值，可以是 true 或 false。
    • Undefined（未定义）：表示一个变量被声明了但是没有赋值。
    • Null（空值）：表示一个空指针，即没有对象。
    • Symbol（符号）：表示唯一且不可变的值，可以用来创建不可重复的属性名。
    • BigInt（大整数）：表示整数，可以表示超过 JavaScript 的最大安全整数。结尾处加上 n：123n。
  • 复杂数据类型（也称为引用类型）：
    • Object（对象）：表示一个键值对集合，可以包含多个属性。
    • Array（数组）：表示一个有序列表，可以包含多个元素。
    • Function（函数）：表示一个可执行的代码块，可以接收参数并返回结果。
  • 基础类型存放于栈，变量记录原始值；引用类型存放堆，变量记录堆地址。

问题：JavaScript 变量再内存中的堆栈存储 ​

js
function fn(obj){
  obj = { m:50 }
  console.log(obj.m, 'fn 内') //50 'fn 内'
}
const o = { m:30 }
fn(o)
console.log(o.m, '全局')  //30 '全局'
  • 解析：
    • 当执行 const o = { m:30 } 时，相当于在堆内存开辟一块空间，存储{ m:30 },同时利用变量 o 记录该堆内存地址，o 存放在栈
    • 当执行 fn(o) 时，会把 o 记录的地址值作为实参传递到方法 fn 中，同时记录在 obj 副本变量中（）
    • 在 fn 函数内部，执行 obj = { m:50 } 时，会在堆内存中新开辟一块空间存储 { m:50 }，并将 obj 变量指向这个新的内存地址
    • 此时 obj 和 o 指向的是不同的堆内存空间，所以 obj.m 输出 50，而原始对象 o.m 依然是 30
    • 当函数执行完毕后，obj 变量被销毁，但原始的 o 对象保持不变，所以全局输出 o.m 仍然是 30 图示说明：
栈内存                 堆内存

                           ↓
+--------+          +-------------+
|  obj   |--+       |  { m:30 }  |

            |       
            |       +-------------+
            +------>|  { m:50 }  | // 赋值后指向新对象

问题：JS单线程设计的目的 ​

javascript 是浏览器的脚本语言，主要用途是以及，多线程编程通常会引发、和等问题。如果以多线程的方式进行浏览器操作，则可能出现不可预测的冲突。假设有两个线程同时操作同一个 DOM 元素，线程1 要求浏览器修改 DOM 内容，而线程2却要求删除 DOM，浏览器就疑惑，无法决定采用哪个线程的操作。所以 JavaScript 的单线程设计很好的简化了这类并发问题，避免了因多线程而引发的竞态条件、死锁和资源竞争等问题。当然，如果在开发中确切需要到异步场景，javascript 也有众多的异步队列来帮助我们实现，也就是我们熟知的事件循环，微任务队列，宏任务队列。如果真的需要开辟一个新线程处理逻辑，也可以通过 webworker 实现。

问题：如何判断对象相等 ​

  1. 浅比较
js
// 方法1：Object.is()
Object.is({}, {})  // false
Object.is([1], [1])  // false

// 方法2：递归比较属性
function shallowEqual(obj1, obj2) {
  if (obj1 === obj2) return true
  
  if (typeof obj1 !== 'object' || obj1 === null ||
      typeof obj2 !== 'object' || obj2 === null) {
    return false
  }
  
  const keys1 = Object.keys(obj1)
  const keys2 = Object.keys(obj2)
  
  if (keys1.length !== keys2.length) return false
  
  return keys1.every(key => 
    obj2.hasOwnProperty(key) && obj1[key] === obj2[key]
  )
}
  1. 深比较
js
// 方法1：JSON.stringify()
JSON.stringify({a:1}) === JSON.stringify({a:1})  // true
// 缺点：无法处理循环引用、函数、undefined、Symbol等

// 方法2：递归深比较
function deepEqual(obj1, obj2) {
  if (obj1 === obj2) return true
  
  if (typeof obj1 !== 'object' || obj1 === null ||
      typeof obj2 !== 'object' || obj2 === null) {
    return false
  }
  
  const keys1 = Object.keys(obj1)
  const keys2 = Object.keys(obj2)
  
  if (keys1.length !== keys2.length) return false
  
  return keys1.every(key => {
    if (!obj2.hasOwnProperty(key)) return false
    return deepEqual(obj1[key], obj2[key])
  })
}

问题：用 setTimeout 来实现倒计时，与 setInterval 的区别 ​

  1. 使用 setTimeout 实现倒计时
js
function countdown(seconds) {
  const start = Date.now()
  
  function tick() {
    const remaining = seconds - Math.floor((Date.now() - start) / 1000)
    
    if (remaining <= 0) {
      console.log('倒计时结束！')
      return
    }
    
    console.log(remaining + '秒')
    setTimeout(tick, 1000)
  }
  
  tick()
}

countdown(5)
  1. setInterval 实现
js
function countdown(seconds) {
  const timer = setInterval(() => {
    seconds--
    
    if (seconds <= 0) {
      clearInterval(timer)
      console.log('倒计时结束！')
      return
    }
    
    console.log(seconds + '秒')
  }, 1000)
}

countdown(5)
  1. 两者区别

  • 执行时机

    • setInterval：固定间隔时间执行，不考虑代码执行时间
    • setTimeout：等待代码执行完后，再等待指定时间执行下一次

  • 精确度

    • setInterval：可能出现跳过或堆叠（如果代码执行时间超过间隔时间）
    • setTimeout：相对更精确，因为每次执行完才计时

  • 内存管理

    • setInterval：需要手动清除定时器
    • setTimeout：递归调用方式，执行完自动结束
  1. 最佳实践建议
js
// 推荐：结合 requestAnimationFrame 实现更精确的定时器
function accurateTimer(fn, delay) {
  const start = performance.now()
  let tickCount = 0
  
  function tick() {
    const elapsed = performance.now() - start
    const targetTick = Math.floor(elapsed / delay)
    
    if (targetTick > tickCount) {
      tickCount = targetTick
      fn()
    }
    
    requestAnimationFrame(tick)
  }
  
  requestAnimationFrame(tick)
}

  })

  .then(() => {
    console.log(1)

// console.log(4) 执行
// console.log(5) 被推入微任务队列

// 第五轮微任务
// console.log(5) 执行
  1. 为什么会这样？
  • Promise.resolve() 会创建一个新的 Promise 实例
  • 每个 then 的回调都会被推入下一轮微任务
  • 当 return 一个 Promise 时，外层 Promise 的状态要等待内层 Promise 状态改变后才会改变
  • 这就导致了 return Promise.resolve(4) 会比普通的值多延迟一轮微任务

进阶：Promise的状态一旦改变就不能再变 ​

js
const first = () => (new Promise((resolve, reject) => {
  console.log(3)                    // 1️⃣ 同步代码
  let p = new Promise((resolve, reject) => {
    console.log(7)                  // 2️⃣ 同步代码
    setTimeout(() => {              // 6️⃣ 宏任务
      console.log(5)
      resolve(6)                    // 无效 resolve，p 已经 resolved
      console.log(p)                // 输出 Promise {<fulfilled>: 1}
    }, 0)
    resolve(1)                      // 3️⃣ 同步代码，p 状态变为 resolved
  })
  resolve(2)                        // 4️⃣ 同步代码，外层 Promise 状态变为 resolved
  p.then(res => {
    console.log(res)                // 5️⃣ 微任务，输出 1
  })
}))

first().then(res => {
  console.log(res)                  // 5️⃣ 微任务，输出 2
})

console.log(4)                      // 1️⃣ 同步代码

// 最终输出顺序：3 7 4 1 2 5 Promise{<fulfilled>: 1}

执行过程分析：

  1. 同步代码执行阶段

    • 输出 3（外层 Promise 的同步代码）
    • 输出 7（内层 Promise p 的同步代码）
    • resolve(1) 将 p 的状态改为 fulfilled
    • resolve(2) 将外层 Promise 的状态改为 fulfilled
    • 输出 4（全局同步代码）

  2. 微任务队列执行阶段

    • 执行 p.then，输出 1
    • 执行 first().then，输出 2

  3. 宏任务队列执行阶段

    • 执行 setTimeout 回调
    • 输出 5
    • resolve(6) 无效，因为 p 已经 resolved
    • 输出 p 的状态，是一个值为 1 的已完成 Promise

关键点解释：

  1. Promise 的状态一旦改变就不能再变
  2. Promise 的 executor 函数是同步执行的
  3. then 回调是微任务
  4. setTimeout 回调是宏任务

  console.log(a)                         // 7️⃣ b 完成后执行 此时 a 是 pending 状态的 Promise
  console.log(3)                         // 8️⃣ 紧接着执行
  await a                                // 9️⃣ 死锁！a 等待自己完成

  console.log(4)                         // 永远不会执行

关键点解释：

  1. async 函数的执行会立即返回一个 Promise
  2. await 会暂停当前 async 函数的执行，等待 Promise 完成

  • 但 await a 需要等待 Promise a 完成

  • resolve(true) 永远不会被执行

问题：什么是内存泄漏 ​

内存泄漏指程序中已分配的内存由于某种原因未被释放或无法释放，造成系统内存的浪费。

常见的内存泄漏情况：

  1. 全局变量
js

  bar = "全局变量"  // 没有声明关键字，意外创建全局变量
  this.variable = "全局变量"  // this 指向 window
}
  1. 闭包使用不当
js

  const someData = new Array(1000000)  // 大数组
  return function inner() {
    // 内部函数引用了外部变量，someData 无法被释放
    return someData
  }
}
  1. 定时器未清除
js

  const someResource = new Array(1000000)
  setInterval(() => {
    // 定时器回调引用了 someResource
    console.log(someResource)
  }, 1000)
  // 没有调用 clearInterval，someResource 无法被释放
}
  1. DOM 引用
js

  button: document.getElementById('button')
}
function removeButton() {
  document.body.removeChild(elements.button)
  // elements.button 仍然保留着对 DOM 元素的引用
}
  1. 事件监听器未移除
js
function addHandler() {
  const el = document.getElementById('el')
  el.addEventListener('click', () => {
    // 处理点击事件
  })
  // 元素被移除时若未移除事件监听器，会造成内存泄漏
}

避免内存泄漏的方法：

  1. 使用 let/const 声明变量
  2. 及时清除定时器
  3. 及时移除事件监听器
  4. 解除不再使用的 DOM 引用
  5. 使用弱引用（WeakMap/WeakSet）

  delete context[fn]
  return result
}

  context = context || window

  context[fn] = this

  return result
}

}
