使用 YouTube Player API ​

首先，我们使用 YouTube Player API 来加载视频，这样我们就可以获取一些基本信息和控制视频播放。

js
<!DOCTYPE html>
<html>

<body>
  <!-- 1. The <iframe> (and video player) will replace this <div> tag. -->
  <div id="player"></div>

  <script>
    // 2. This code loads the IFrame Player API code asynchronously.
    var tag = document.createElement('script');
    tag.src = "https://www.youtube.com/iframe_api";
    var firstScriptTag = document.getElementsByTagName('script')[0];
    firstScriptTag.parentNode.insertBefore(tag, firstScriptTag);

    // 3. This function creates an <iframe> (and YouTube player)
    //    after the API code downloads.
    var player;
    function onYouTubeIframeAPIReady() {
      player = new YT.Player('player', {
        height: '360',
        width: '640',
        videoId: 'M7lc1UVf-VE',
        events: {
          'onReady': onPlayerReady,
          'onStateChange': onPlayerStateChange
        }
      });
    }

    // 4. The API will call this function when the video player is ready.
    function onPlayerReady(event) {
      event.target.playVideo();
      getVideoInfo();
    }

    // 5. The API calls this function when the player's state changes.
    function onPlayerStateChange(event) {
      if (event.data == YT.PlayerState.PLAYING) {
        // Video is playing
      }
    }

    function getVideoInfo() {
      // 获取视频信息
      var videoData = player.getVideoData();
      console.log("Video Data:", videoData);
      console.log("Title:", videoData.title);
      console.log("Video ID:", videoData.video_id);
      console.log("Author:", videoData.author);

      // 获取视频时长（秒）
      var duration = player.getDuration();
      console.log("Duration:", duration);

      // 获取缩略图
      var videoId = player.getVideoData().video_id;
      var thumbnailUrl = `https://img.youtube.com/vi/${videoId}/0.jpg`;
      console.log("Thumbnail URL:", thumbnailUrl);
    }
  </script>

async function scrapeData(urls) {
  const browser = await puppeteer.launch({
    headless: "new",
    args: ['--no-sandbox', '--disable-setuid-sandbox']
  });

  const results = [];

  try {
    for (const url of urls) {

        const videoId = new URL(url).searchParams.get('v');

        const videoTitle = ytInitialData.playerOverlays?.playerOverlayRenderer?.videoDetails?.playerOverlayVideoDetailsRenderer?.title?.simpleText;

        let videoDescription = '';
        let authorAvatar = '';
        let authorName = '';
        let authorEmail = '';

        const contents = ytInitialData.contents?.twoColumnWatchNextResults?.results?.results?.contents;
        if (contents) {
          const data = contents.find(item => item.videoSecondaryInfoRenderer);
          if (data) {
            const videoSecondaryInfoRenderer = data.videoSecondaryInfoRenderer;
            videoDescription = videoSecondaryInfoRenderer.attributedDescription?.content || '';
            authorAvatar = videoSecondaryInfoRenderer.owner?.videoOwnerRenderer?.thumbnail?.thumbnails?.[0]?.url || '';
            authorName = videoSecondaryInfoRenderer.owner?.videoOwnerRenderer?.title?.runs?.[0]?.text || '';
            authorEmail = videoSecondaryInfoRenderer.owner?.videoOwnerRenderer?.title?.runs?.[0]?.text?.navigationEndpoint?.commandMetadata?.webCommandMetadata?.url || '';
            thumbnailUrl = `https://img.youtube.com/vi/${videoId}/0.jpg`
          }
        }

        return {
          videoId,
          videoTitle,
          videoDescription,
          authorAvatar,
          authorName,
          authorEmail,
          thumbnailUrl
        };
      }, url);

      if (extractedData) {
        console.log(`Successfully extracted data for video ID: ${extractedData.videoId}`);

        // 调用 oEmbed API
        const oembedUrl = `https://www.youtube.com/oembed?url=${encodeURIComponent(url)}&format=json`;
        try {
          const response = await fetch(oembedUrl);
          const oembedData = await response.json();

          oembedData.id = extractedData.videoId;
          const email = extractUsernameFromUrl(oembedData.author_url);
          oembedData.email = "@" + email;

          // 合并数据
          const mergedData = {
            ...extractedData,
            ...oembedData,
            thumbnail_url: oembedData.thumbnail_url,
            title: oembedData.title,
            author_url: oembedData.author_url
          };

          results.push(mergedData);
        } catch (error) {
          console.error('Error fetching oEmbed data:', error);
          results.push(extractedData);
        }
      } else {
        console.log(`Failed to extract data for URL: ${url}`);
      }

      await page.close();
    }

    // 将所有结果保存为 JSON 文件
    fs.writeFileSync('extractedVideosData.json', JSON.stringify(results, null, 2));
    console.log('All data saved to extractedVideosData.json');

  } catch (error) {
    console.error('Error scraping data:', error);
  } finally {
    await browser.close();
  }

  return results;
}

function determineType(url) {
  const typeMap = {
    'youtube.com': 'youtube',
    'twitter.com': 'x',
    'x.com': 'x',
    'bilibili.com': 'bilibili',
    'xiaohongshu.com': 'xiaohongshu',
    "facebook.com": 'facebook'
  };

  for (const domain in typeMap) {
    if (url.includes(domain)) {
      return typeMap[domain];
    }
  }

  return 'unknown';
}

//bilibili图片直接引用会跨域，保存到本地 evaluate是浏览器环境，无法使用path
async function downloadAndSaveImage(url, prefix, width = 240, height = 240) {
  if (!url) return null;
  const cleanUrl = url.split('@')[0];  // 移除尺寸参数
  const hash = crypto.createHash('md5').update(cleanUrl).digest('hex');
  const filename = `${prefix}_${hash}.jpg`;
  const filepath = path.join(__dirname, 'public', 'assets', 'storydata', 'bilibili', filename);

  try {
    // 下载原始图片
    const response = await axios({
      url: cleanUrl,
      method: 'GET',
      responseType: 'stream'
    });

    // 使用 sharp 调整尺寸
    await new Promise((resolve, reject) => {
      response.data
        .pipe(sharp())
        .resize(width, height) // 调整尺寸
        .toFile(filepath, (err, info) => {
          if (err) {
            reject(err);
          } else {
            resolve(info);
          }
        });
    });

    return `/assets/storydata/bilibili/${filename}`;
  } catch (error) {
    console.error(`Error downloading and resizing image from ${url}:`, error);
    return null;
  }
}


function parseDate(dateString) {
  const formats = [
    /(\d{4})年(\d{1,2})月(\d{1,2})日/, // "2024年6月12日"
    /(\d{4})-(\d{1,2})-(\d{1,2})/, // "2024-06-18"
    /(\d{1,2}):(\d{1,2}) · (\d{4})年(\d{1,2})月(\d{1,2})日/, // "上午10:59 · 2024年8月20日"
    /(\d{2})-(\d{2})/, // "07-29"
  ];

  for (const format of formats) {
    console.log(dateString, 'dateString');
    if (dateString) {
      const match = dateString.match(format);
      if (match) {
        let year, month, day
        if (match.length === 3) {
          year = new Date().getFullYear()
          month = parseInt(match[1], 10) - 1;
          day = parseInt(match[2], 10);
        } else {
          year = parseInt(match[1], 10);
          month = parseInt(match[2], 10) - 1; // JavaScript 月份是从 0 开始的
          day = parseInt(match[3], 10);
        }
        return new Date(year, month, day);
      }
    }
  }
  return ''
}

//格式化时间并排序
function sortByDate(arr, descending = true) {
  return arr.map(item => {
    const parsedDate = parseDate(item.date);
    if (parsedDate) {
      item.formattedDate = parsedDate?.toISOString()?.substring(0, 10);
    }
    return item;
  }).sort((a, b) => {
    const dateA = new Date(a.formattedDate);
    const dateB = new Date(b.formattedDate);

    if (descending) {
      return dateB - dateA; // 降序
    } else {
      return dateA - dateB; // 升序
    }
  });
}

function clearDirectory(directoryPath) {
  fs.readdir(directoryPath, (err, files) => {
    if (err) {
      console.error(`Error reading directory: ${directoryPath}`, err);
      return;

function parseByBracketMatching(scriptContent, siteConfig) {
  const dataKeyIndex = scriptContent.indexOf(`,"${siteConfig.dataKey}":`);
  if (dataKeyIndex === -1) return;
  const dataStartIndex = scriptContent.indexOf('{', dataKeyIndex);
  if (dataStartIndex === -1) return;
  const stack = ['{'];
  let dataEndIndex = dataStartIndex + 1;

  while (dataEndIndex < scriptContent.length && stack.length > 0) {
    const char = scriptContent[dataEndIndex];
    if (char === '{') stack.push('{');
    else if (char === '}') stack.pop();
    dataEndIndex++;
  }

  if (stack.length === 0) {
    const dataStr = scriptContent.substring(dataStartIndex, dataEndIndex);
    return JSON.parse(dataStr);
  }
}

async function crawlVideoByHtml(videoUrl) {
  try {
    const { data: html } = await axios.get(videoUrl);
    const $ = cheerio.load(html);

    const siteConfig = Object.values(SITE_CONFIGS).find(config =>
      videoUrl.includes(config.identifier)
    );

    if (!siteConfig) {
      throw new Error('不支持的网站');
    }

    let videoData = null;
    $('script').each((index, element) => {
      const scriptContent = $(element).html();
      if (!scriptContent) return;

      // 遍历所有解析策略
      for (const strategy of siteConfig.parseStrategies) {
        try {
          if (strategy.pattern) {
            const match = scriptContent.match(strategy.pattern);
            if (match) {
              videoData = strategy.extract(match, siteConfig);
              if (videoData) {
                console.log(`使用 ${strategy.name} 策略成功解析数据 --- `);
                return false;
              }
            }
          } else {
            videoData = strategy.extract(scriptContent, siteConfig);
            if (videoData) {
              console.log(`使用 ${strategy.name} 策略成功解析数据`);
              return false;
            }
          }
        } catch (err) {
          console.error(`${strategy.name} 策略解析失败:`, err);
          continue;
        }
      }
    });

    return videoData;
  } catch (error) {
    console.error('爬取失败:', error);
    throw error;
  }
}
