🏗️ 项目架构概览 ​

在上一篇文章中，我们完成了基础环境的搭建。现在让我们深入了解如何构建一个完整的Express增删改查（CRUD）系统。本项目采用分层架构设计，遵循单一职责原则，具备良好的可维护性和扩展性。

📁 项目目录结构解析 ​

project/
├── app.js                 # 应用程序入口文件

│   ├── index.js           # 模型初始化和关联配置

    └── middleware.js      # 通用中间件函数

// models/article.js
'use strict';
const { Model } = require('sequelize');

module.exports = (sequelize, DataTypes) => {
  class Article extends Model {
    static associate(models) {
      // 在这里定义模型关联关系
      // 例如：Article.belongsTo(models.User, { foreignKey: 'userId' });
    }
  }
  
  Article.init({
    title: {
      type: DataTypes.STRING,
      allowNull: false,
      validate: {
        notNull: { msg: 'Title is required' },
        notEmpty: { msg: 'Title is required' },
        len: {
          args: [2, 30],
          msg: 'Title must be between 2 and 30 characters'
        }
      }
    },
    content: {
      type: DataTypes.TEXT,
      allowNull: false,
      validate: {
        notNull: { msg: 'Content is required' },
        notEmpty: { msg: 'Content is required' },
        len: {
          args: [1, 1000],
          msg: 'Content must be between 1 and 1000 characters'
        }
      }
    }
  }, {
    sequelize,
    modelName: 'Article',
    // 自动添加 createdAt 和 updatedAt 字段
    timestamps: true
  });
  
  return Article;
};

🛠️ 业务逻辑层（Services） ​

Service层是整个应用的核心，负责处理所有业务逻辑。让我们详细分析文章服务的实现：

📄 ArticleService类设计 ​

javascript
// services/articleService.js
const { Article } = require('../models');
const { Op } = require('sequelize');
const { NotFoundError } = require('../utils/response');

class ArticleService {
  /**
   * 获取文章列表（支持搜索和分页）
   * @param {Object} params - 查询参数
   * @param {string} params.title - 标题搜索关键词
   * @param {string} params.content - 内容搜索关键词  
   * @param {Object} params.pagination - 分页参数
   * @returns {Object} 文章列表和分页信息
   */
  static async getArticles({ title, content, pagination }) {
    const { currentPage, pageSize } = pagination;

    // 构建查询条件
    const condition = {
      where: {},
      order: [['id', 'DESC']],          // 按ID降序排列
      limit: pageSize,                   // 限制返回数量
      offset: (currentPage - 1) * pageSize  // 计算偏移量
    };

    // 动态构建搜索条件
    if (title || content) {
      condition.where = { [Op.or]: [] };

      if (title) {
        condition.where[Op.or].push({ 
          title: { [Op.like]: `%${title}%` } 
        });
      }

      if (content) {
        condition.where[Op.or].push({ 
          content: { [Op.like]: `%${content}%` } 
        });
      }
    }

    // 执行查询并获取总数
    const { rows: articles, count: total } = await Article.findAndCountAll(condition);

    return {
      articles,
      pagination: {
        currentPage,
        pageSize,
        total,
        totalPages: Math.ceil(total / pageSize)
      }
    };
  }

  /**
   * 根据ID获取单个文章
   * @param {number} id - 文章ID
   * @returns {Object} 文章对象
   * @throws {NotFoundError} 文章不存在时抛出错误
   */
  static async getArticleById(id) {
    const article = await Article.findByPk(id);
    if (!article) {
      throw new NotFoundError('Article not found');
    }
    return article;
  }

  /**
   * 创建新文章
   * @param {Object} articleData - 文章数据
   * @returns {Object} 创建的文章对象
   */
  static async createArticle(articleData) {
    // 过滤只允许的字段，防止恶意注入
    const filteredData = this.filterAllowedFields(articleData);
    return await Article.create(filteredData);
  }

  /**
   * 更新文章
   * @param {number} id - 文章ID
   * @param {Object} articleData - 更新的文章数据
   * @returns {Object} 更新后的文章对象
   */
  static async updateArticle(id, articleData) {
    const article = await this.getArticleById(id);
    const filteredData = this.filterAllowedFields(articleData);
    await article.update(filteredData);
    return article;
  }

  /**
   * 删除文章
   * @param {number} id - 文章ID
   * @returns {Object} 被删除的文章对象
   */
  static async deleteArticle(id) {
    const article = await this.getArticleById(id);
    await article.destroy();
    return article;
  }

  /**
   * 字段过滤器 - 安全性考虑
   * @param {Object} data - 原始数据
   * @returns {Object} 过滤后的数据
   */
  static filterAllowedFields(data) {
    const allowedFields = ['title', 'content'];
    return Object.fromEntries(
      Object.entries(data).filter(([key]) => allowedFields.includes(key))
    );
  }
}

module.exports = ArticleService;

  2. 错误处理：统一的错误抛出和处理机制

📤 统一响应格式 ​

javascript
// utils/response.js

// 自定义错误类
class NotFoundError extends Error {
  constructor(message) {
    super(message);
    this.name = 'NotFoundError';
    this.statusCode = 404;
  }
}

class ValidationError extends Error {
  constructor(message, errors = []) {
    super(message);
    this.name = 'ValidationError';
    this.statusCode = 400;
    this.errors = errors;
  }
}

/**
 * 成功响应格式
 * @param {Object} res - Express响应对象
 * @param {string} message - 响应消息
 * @param {*} data - 响应数据
 * @param {number} code - HTTP状态码
 */
function success(res, message, data = {}, code = 200) {
  res.status(code).json({
    status: true,
    message,
    data,
    timestamp: new Date().toISOString()
  });
}

/**
 * 分页响应格式
 * @param {Object} res - Express响应对象
 * @param {string} message - 响应消息
 * @param {Array} data - 数据数组
 * @param {Object} pagination - 分页信息
 */
function successWithPagination(res, message, data, pagination) {
  res.status(200).json({
    status: true,
    message,
    data,
    pagination,
    timestamp: new Date().toISOString()
  });
}

/**
 * 错误响应处理
 * @param {Object} res - Express响应对象
 * @param {Error} error - 错误对象
 */
function failure(res, error) {
  let statusCode = 500;
  let message = error.message;
  let errors = [];

  // 根据错误类型设置不同的响应
  if (error.name === 'SequelizeValidationError') {
    statusCode = 400;
    message = 'Validation failed';
    errors = error.errors.map(e => e.message);
  } else if (error.name === 'NotFoundError') {
    statusCode = 404;
  } else if (error.name === 'ValidationError') {
    statusCode = 400;
    errors = error.errors || [];
  }

  res.status(statusCode).json({
    status: false,
    message,
    errors: errors.length > 0 ? errors : undefined,
    timestamp: new Date().toISOString()
  });
}

🔧 中间件函数集合 ​

javascript
// utils/middleware.js
const { failure } = require('./response');

/**
 * 异步路由错误处理包装器
 * 自动捕获async/await中的错误并传递给错误处理中间件
 */
const asyncHandler = (fn) => {
  return (req, res, next) => {
    Promise.resolve(fn(req, res, next)).catch(next);
  };
};

/**
 * 全局错误处理中间件
 * 统一处理所有未捕获的错误
 */
const errorHandler = (err, req, res, next) => {
  console.error('错误详情:', err);
  failure(res, err);
};

/**
 * 分页参数处理中间件
 * 解析和验证分页参数
 */
const paginationMiddleware = (req, res, next) => {
  const currentPage = parseInt(req.query.currentPage) || 1;
  const pageSize = parseInt(req.query.pageSize) || 10;

  // 设置合理的分页限制
  req.pagination = {
    currentPage: Math.max(1, currentPage),
    pageSize: Math.min(Math.max(1, pageSize), 100), // 最大100条
  };

  next();
};

/**
 * 参数验证中间件生成器
 * @param {Object} validationRules - 验证规则对象
 * @returns {Function} 中间件函数
 */
const validateParams = (validationRules) => {
  return (req, res, next) => {
    const errors = [];

    for (const [field, rules] of Object.entries(validationRules)) {
      const value = req.body[field] || req.query[field] || req.params[field];

      // 必填字段验证
      if (rules.required && (!value || value.trim() === '')) {
        errors.push(`${field} is required`);
        continue;
      }

      // 最小长度验证
      if (value && rules.minLength && value.length < rules.minLength) {
        errors.push(`${field} must be at least ${rules.minLength} characters`);
      }

      // 最大长度验证
      if (value && rules.maxLength && value.length > rules.maxLength) {
        errors.push(`${field} must not exceed ${rules.maxLength} characters`);
      }
    }

    if (errors.length > 0) {
      return res.status(400).json({
        status: false,
        message: 'Validation failed',
        errors
      });
    }

    next();
  };
};

🛣️ 路由控制层（Routes） ​

路由层是客户端与服务器交互的入口，负责接收HTTP请求并调用相应的业务逻辑。

📝 文章CRUD路由实现 ​

javascript
// routes/admin/articles.js
const express = require('express');
const router = express.Router();
const ArticleService = require('../../services/articleService');
const { success, successWithPagination } = require('../../utils/response');
const { asyncHandler, paginationMiddleware, validateParams } = require('../../utils/middleware');

// 验证规则定义
const createArticleValidation = {
  title: { required: true, minLength: 2, maxLength: 30 },
  content: { required: true, minLength: 1, maxLength: 1000 }
};

const updateArticleValidation = {
  title: { minLength: 2, maxLength: 30 },
  content: { minLength: 1, maxLength: 1000 }
};

/**
 * 获取文章列表（支持搜索和分页）
 * GET /admin/articles?title=xxx&content=xxx&currentPage=1&pageSize=10
 */
router.get('/', paginationMiddleware, asyncHandler(async (req, res) => {
  const { title, content } = req.query;
  const { pagination } = req;

  const result = await ArticleService.getArticles({ title, content, pagination });

  successWithPagination(res, 'Articles fetched successfully', result.articles, result.pagination);
}));

/**
 * 获取单个文章
 * GET /admin/articles/:id
 */
router.get('/:id', asyncHandler(async (req, res) => {
  const { id } = req.params;
  const article = await ArticleService.getArticleById(id);

  success(res, 'Article fetched successfully', article);
}));

/**
 * 创建新文章
 * POST /admin/articles
 */
router.post('/', validateParams(createArticleValidation), asyncHandler(async (req, res) => {
  const article = await ArticleService.createArticle(req.body);

  success(res, 'Article created successfully', article, 201);
}));

/**
 * 更新文章
 * PUT /admin/articles/:id
 */
router.put('/:id', validateParams(updateArticleValidation), asyncHandler(async (req, res) => {
  const { id } = req.params;
  const article = await ArticleService.updateArticle(id, req.body);

  success(res, 'Article updated successfully', article);
}));

/**
 * 删除文章
 * DELETE /admin/articles/:id
 */
router.delete('/:id', asyncHandler(async (req, res) => {
  const { id } = req.params;
  await ArticleService.deleteArticle(id);

  success(res, 'Article deleted successfully', null);
}));

module.exports = router;

🎯 核心特性与亮点 ​

1. 分层架构设计 🏗️ ​

  • 职责分离：每层都有明确的职责边界
  • 松耦合：层与层之间通过接口交互，便于测试和维护
  • 高内聚：相关功能聚合在同一层内

2. 统一响应格式 📤 ​

  • 标准化：所有API返回统一的JSON格式
  • 错误处理：完善的错误分类和错误信息
  • 时间戳：每个响应都包含时间戳信息

3. 参数验证机制 🛡️ ​

  • 多层验证：路由层和模型层双重验证
  • 自定义验证：灵活的验证规则配置
  • 友好提示：清晰的错误提示信息

4. 分页和搜索 🔍 ​

  • 性能优化：合理的分页机制防止大数据量查询
  • 模糊搜索：支持标题和内容的模糊匹配
  • 参数限制：防止恶意的大分页请求

5. 安全性考虑 🔒 ​

  • 字段过滤：只允许指定字段的更新
  • SQL注入防护：使用ORM参数化查询

static async getPopularArticles() {

  const cached = await redis.get(cacheKey);
  
  if (cached) {
    return JSON.parse(cached);
  }
  
  const articles = await Article.findAll({
    limit: 10,
    order: [['views', 'DESC']]
  });
  
  await redis.setex(cacheKey, 3600, JSON.stringify(articles));
  return articles;
}
