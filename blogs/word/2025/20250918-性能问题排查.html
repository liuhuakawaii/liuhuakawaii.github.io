前端性能问题排查实战指南：从发现到解决的完整流程 🚀 ​

🎯 前言 ​

"性能优化不是一次性的任务，而是一个持续的过程。"

在现代前端开发的复杂生态中，性能问题如同隐藏在代码深处的幽灵，往往在用户体验最关键的时刻现身。一个看似简单的页面卡顿，背后可能交织着内存泄漏的蛛网、DOM操作的风暴、以及事件监听器的幽魂。

作为前端工程师，我们不仅要会写功能，更要具备性能侦探的敏锐嗅觉。本文将通过真实生产环境案例，为你构建一套科学系统的性能问题排查方法论——从蛛丝马迹的发现、抽丝剥茧的分析，到药到病除的解决，每一个环节都配备专业工具和实战技巧。

🎯 问题识别的关键指标 ​

用户感知层面：

  • 页面响应时间 > 100ms（用户开始感知延迟）
  • 首屏渲染时间 > 2.5s（用户开始流失）
  • 内存占用持续增长（浏览器最终崩溃）

技术指标层面：

  • FCP (First Contentful Paint) > 1.8s
  • LCP (Largest Contentful Paint) > 2.5s
  • CLS (Cumulative Layout Shift) > 0.1
  • FID (First Input Delay) > 100ms

1.2 快速诊断工具包 🛠️ ​

🌳 DOM节点泄漏监控器

js
/**
 * DOM节点泄漏检测器
 * 用途：实时监控页面DOM节点数量变化，及时发现节点泄漏问题
 * 使用：在控制台运行，观察节点数量是否异常增长
 */
class DOMLeakDetector {
  constructor(options = {}) {
    this.interval = options.interval || 2000;
    this.threshold = options.threshold || 1000; // 脱离节点阈值
    this.history = [];
    this.timer = null;
  }

  start() {
    this.timer = setInterval(() => {
      const stats = this.getNodeStats();
      this.history.push({ ...stats, timestamp: Date.now() });
      
      // 保持最近10次记录
      if (this.history.length > 10) {
        this.history.shift();
      }
      
      this.logStats(stats);
      this.checkLeaks(stats);
    }, this.interval);
    
    console.log('🔍 DOM泄漏监控已启动...');
  }

  stop() {
    if (this.timer) {
      clearInterval(this.timer);
      this.timer = null;
      console.log('⏹️ DOM泄漏监控已停止');
    }
  }

  getNodeStats() {
    const allNodes = document.getElementsByTagName('*');
    const detachedNodes = Array.from(allNodes).filter(n => !n.isConnected);
    
    return {
      total: allNodes.length,
      detached: detachedNodes.length,
      ratio: (detachedNodes.length / allNodes.length * 100).toFixed(2)
    };
  }

  logStats(stats) {
    const trend = this.getTrend();
    console.log(`🌳 [DOM监控] 总节点: ${stats.total} | 脱离节点: ${stats.detached} (${stats.ratio}%) ${trend}`);
  }

  getTrend() {
    if (this.history.length < 2) return '';
    const current = this.history[this.history.length - 1];
    const previous = this.history[this.history.length - 2];
    const change = current.detached - previous.detached;
    
    if (change > 0) return `📈 +${change}`;
    if (change < 0) return `📉 ${change}`;
    return '➡️ 稳定';
  }

  checkLeaks(stats) {
    if (stats.detached > this.threshold) {
      console.warn(`⚠️ 检测到可能的DOM泄漏！脱离节点数量: ${stats.detached}`);
    }
  }
}

// 使用示例
window.domDetector = new DOMLeakDetector({ interval: 3000, threshold: 500 });
window.domDetector.start();

// 停止监控：window.domDetector.stop();

🧠 深度解析：脱离节点（Detached Nodes） ​

什么是脱离节点？

脱离节点是前端内存泄漏的重要信号。简单来说，当一个DOM节点已经从文档树中移除，但JavaScript代码中仍然持有对它的引用时，这个节点就成为了"脱离节点"。

mermaid
graph TD
    A[DOM节点] --> B{是否从文档中移除?}
    B -->|是| C{JavaScript是否还有引用?}
    B -->|否| D[正常节点]
    C -->|是| E[脱离节点 ⚠️]
    C -->|否| F[可被GC回收 ✅]

🔍 脱离节点产生的典型场景

场景1：变量持有DOM引用

js
// ❌ 错误示例：变量持有已移除节点的引用
let globalDiv;

function createAndRemoveNode() {
  globalDiv = document.createElement('div');
  globalDiv.innerHTML = '大量内容...';
  document.body.appendChild(globalDiv);
  
  // 1秒后移除节点
  setTimeout(() => {
    document.body.removeChild(globalDiv);
    // 🚨 问题：globalDiv变量仍然引用着这个节点！
  }, 1000);
}

// ✅ 正确做法：及时清理引用
function createAndRemoveNodeCorrect() {
let div = document.createElement('div');
document.body.appendChild(div);
  
  setTimeout(() => {
    document.body.removeChild(div);
    div = null; // 清理引用，让GC可以回收
  }, 1000);
}

场景2：事件监听器未解绑

js
// ❌ 错误示例：事件监听器导致的脱离节点
function createButtonWithLeak() {
  const button = document.createElement('button');
  const largeData = new Array(100000).fill('data'); // 大对象
  
  // 事件处理函数形成闭包，引用了largeData
  button.addEventListener('click', () => {
    console.log('处理大量数据:', largeData.length);
  });
  
  document.body.appendChild(button);
  
  // 移除按钮，但事件监听器仍然存在！
  setTimeout(() => {
    document.body.removeChild(button);
    // 🚨 button + largeData 都无法被回收
  }, 2000);
}

// ✅ 正确做法：移除前先解绑事件
function createButtonCorrect() {
  const button = document.createElement('button');
  const largeData = new Array(100000).fill('data');
  
  const handleClick = () => {
    console.log('处理大量数据:', largeData.length);
  };
  
  button.addEventListener('click', handleClick);
  document.body.appendChild(button);
  
  setTimeout(() => {
    button.removeEventListener('click', handleClick); // 先解绑
    document.body.removeChild(button); // 再移除
  }, 2000);
}

场景3：React/Vue组件中的常见陷阱

jsx
// ❌ React中的脱离节点陷阱
function ProblematicComponent() {
  const [nodes, setNodes] = useState([]);
  
  useEffect(() => {
    const interval = setInterval(() => {
      const div = document.createElement('div');
      document.body.appendChild(div);
      
      // 🚨 将DOM节点存储在state中
      setNodes(prev => [...prev, div]);
      
      // 稍后移除节点，但state中仍有引用
      setTimeout(() => {
        document.body.removeChild(div);
        // nodes数组仍然引用着这些已移除的节点！
      }, 1000);
    }, 500);
    
    return () => clearInterval(interval);
  }, []);
  
  return <div>组件内容</div>;
}

// ✅ 正确的处理方式
function CorrectComponent() {
  const nodeRefs = useRef(new Set());
  
  useEffect(() => {
    const interval = setInterval(() => {
      const div = document.createElement('div');
  document.body.appendChild(div);
      nodeRefs.current.add(div);
      
      setTimeout(() => {
  document.body.removeChild(div);
        nodeRefs.current.delete(div); // 清理引用
      }, 1000);
    }, 500);
    
    return () => {
      clearInterval(interval);
      // 组件卸载时清理所有引用
      nodeRefs.current.forEach(node => {
        if (node.parentNode) {
          node.parentNode.removeChild(node);
        }
      });
      nodeRefs.current.clear();
    };
  }, []);
  
  return <div>组件内容</div>;
}

🧮 JavaScript堆内存指标详解

在进行内存分析时，理解V8引擎的内存管理机制至关重要：

指标含义作用正常范围

📈 高级内存监控器

js
/**
 * 内存泄漏监控系统
 * 功能：实时监控内存使用情况，自动识别异常增长模式
 */
class MemoryLeakDetector {
  constructor(options = {}) {
    this.interval = options.interval || 3000;
    this.alertThreshold = options.alertThreshold || 0.8; // 80%内存使用率告警
    this.growthThreshold = options.growthThreshold || 10; // 10MB增长告警
    this.history = [];
    this.timer = null;
    this.isAlerting = false;
  }

  start() {
    if (!performance.memory) {
      console.warn('⚠️ 当前浏览器不支持 performance.memory API');
      return;
    }

    this.timer = setInterval(() => {
      const memInfo = this.getMemoryInfo();
      this.history.push(memInfo);
      
      // 保持最近20次记录用于趋势分析
      if (this.history.length > 20) {
        this.history.shift();
      }
      
      this.logMemoryStats(memInfo);
      this.analyzeMemoryTrend(memInfo);
    }, this.interval);
    
    console.log('🔍 内存泄漏监控已启动...');
  }

  stop() {
    if (this.timer) {
      clearInterval(this.timer);
      this.timer = null;
      console.log('⏹️ 内存监控已停止');
    }
  }

  getMemoryInfo() {
    const { usedJSHeapSize, totalJSHeapSize, jsHeapSizeLimit } = performance.memory;
    const timestamp = Date.now();
    
    return {
      timestamp,
      used: usedJSHeapSize,
      total: totalJSHeapSize,
      limit: jsHeapSizeLimit,
      usedMB: (usedJSHeapSize / 1024 / 1024).toFixed(2),
      totalMB: (totalJSHeapSize / 1024 / 1024).toFixed(2),
      limitMB: (jsHeapSizeLimit / 1024 / 1024).toFixed(2),
      usageRatio: (usedJSHeapSize / jsHeapSizeLimit).toFixed(3),
      efficiency: (usedJSHeapSize / totalJSHeapSize).toFixed(3)
    };
  }

  logMemoryStats(memInfo) {
    const trend = this.getMemoryTrend();
    const status = this.getMemoryStatus(memInfo);
    
    console.log(
      `💾 [内存监控] ${status} 已用: ${memInfo.usedMB}MB/${memInfo.limitMB}MB ` +
      `(${(memInfo.usageRatio * 100).toFixed(1)}%) | 堆效率: ${(memInfo.efficiency * 100).toFixed(1)}% ${trend}`
    );
  }

  getMemoryStatus(memInfo) {
    const ratio = parseFloat(memInfo.usageRatio);
    if (ratio > 0.9) return '🔴';
    if (ratio > 0.7) return '🟠';
    if (ratio > 0.5) return '🟡';
    return '🟢';
  }

  getMemoryTrend() {
    if (this.history.length < 3) return '';
    
    const recent = this.history.slice(-3);
    const growth = recent[2].used - recent[0].used;
    const growthMB = (growth / 1024 / 1024).toFixed(1);
    
    if (growth > 10 * 1024 * 1024) return `📈 +${growthMB}MB`;
    if (growth < -5 * 1024 * 1024) return `📉 ${growthMB}MB`;
    return '➡️ 稳定';
  }

  analyzeMemoryTrend(memInfo) {
    if (this.history.length < 5) return;
    
    // 检查内存使用率告警
    const ratio = parseFloat(memInfo.usageRatio);
    if (ratio > this.alertThreshold && !this.isAlerting) {
      this.triggerAlert('HIGH_USAGE', `内存使用率过高: ${(ratio * 100).toFixed(1)}%`);
    }
    
    // 检查内存增长趋势
    const recentGrowth = this.calculateRecentGrowth();
    if (recentGrowth > this.growthThreshold && !this.isAlerting) {
      this.triggerAlert('RAPID_GROWTH', `内存快速增长: +${recentGrowth.toFixed(1)}MB`);
    }
    
    // 重置告警状态
    if (ratio < this.alertThreshold - 0.1 && recentGrowth < this.growthThreshold / 2) {
      this.isAlerting = false;
    }
  }

  calculateRecentGrowth() {
    if (this.history.length < 5) return 0;
    
    const recent5 = this.history.slice(-5);
    const startMem = recent5[0].used;
    const endMem = recent5[4].used;
    
    return (endMem - startMem) / 1024 / 1024; // 转换为MB
  }

  triggerAlert(type, message) {
    this.isAlerting = true;
    console.warn(`🚨 [内存告警] ${type}: ${message}`);
    
    // 可以在这里添加更多告警处理逻辑
    // 比如发送监控数据到服务端、显示用户提示等
    if (typeof window !== 'undefined' && window.dispatchEvent) {
      window.dispatchEvent(new CustomEvent('memoryAlert', {
        detail: { type, message, memoryInfo: this.history[this.history.length - 1] }
      }));
    }
  }

  // 获取内存分析报告
  getReport() {
    if (this.history.length === 0) return null;
    
    const latest = this.history[this.history.length - 1];
    const oldest = this.history[0];
    const totalGrowth = (latest.used - oldest.used) / 1024 / 1024;
    const avgGrowthRate = totalGrowth / this.history.length;
    
    return {
      监控时长: `${((latest.timestamp - oldest.timestamp) / 1000 / 60).toFixed(1)}分钟`,
      当前内存使用: `${latest.usedMB}MB`,
      内存使用率: `${(latest.usageRatio * 100).toFixed(1)}%`,
      总增长: `${totalGrowth.toFixed(1)}MB`,
      平均增长率: `${avgGrowthRate.toFixed(2)}MB/次`,
      堆内存效率: `${(latest.efficiency * 100).toFixed(1)}%`,
      告警状态: this.isAlerting ? '🚨 异常' : '✅ 正常'
    };
  }
}

// 使用示例
window.memoryDetector = new MemoryLeakDetector({
  interval: 5000,        // 5秒检测一次
  alertThreshold: 0.75,  // 75%内存使用率告警
  growthThreshold: 15    // 15MB增长告警
});

window.memoryDetector.start();

// 查看报告：console.table(window.memoryDetector.getReport());
// 停止监控：window.memoryDetector.stop();

function createBlobLeak() {

  });
}

/**
 * 主线程阻塞检测器
 * 用途：检测长任务（Long Tasks），识别主线程阻塞问题
 */
class MainThreadBlockageDetector {
  constructor(threshold = 50) {
    this.threshold = threshold; // 超过50ms被认为是长任务
    this.longTasks = [];
    this.observer = null;
  }

  start() {
    // 使用 PerformanceObserver 监控长任务
    if ('PerformanceObserver' in window) {
      this.observer = new PerformanceObserver((list) => {
        for (const entry of list.getEntries()) {
          if (entry.duration > this.threshold) {
            this.recordLongTask(entry);
          }
        }
      });
      
      this.observer.observe({ entryTypes: ['longtask'] });
      console.log('🔍 主线程阻塞监控已启动...');
    }

    // 备用方案：使用 setTimeout 检测阻塞
    this.startTimeoutDetection();
  }

  recordLongTask(entry) {
    const taskInfo = {
      startTime: entry.startTime,
      duration: entry.duration,
      attribution: entry.attribution || [],
      timestamp: Date.now()
    };

    this.longTasks.push(taskInfo);
    
    console.warn(
      `🚨 检测到长任务! 持续时间: ${entry.duration.toFixed(2)}ms, ` +
      `开始时间: ${entry.startTime.toFixed(2)}ms`
    );

    // 如果任务超过100ms，发出严重告警
    if (entry.duration > 100) {
      console.error(
        `💀 严重阻塞! 任务执行了 ${entry.duration.toFixed(2)}ms，` +
        `可能导致页面卡顿和ANR`
      );
    }
  }

  // 备用检测方案：通过setTimeout间隔检测主线程阻塞
  startTimeoutDetection() {
    let lastTime = performance.now();
    
    const checkBlocking = () => {
      const currentTime = performance.now();
      const gap = currentTime - lastTime;
      
      // 如果时间间隔明显超过预期，说明主线程被阻塞
      if (gap > this.threshold + 10) { // +10ms容错
        console.warn(
          `⏰ 主线程阻塞检测: 预期间隔5ms，实际间隔${gap.toFixed(2)}ms`
        );
      }
      
      lastTime = currentTime;
      setTimeout(checkBlocking, 5);
    };
    
    setTimeout(checkBlocking, 5);
  }

  stop() {
    if (this.observer) {
      this.observer.disconnect();
      console.log('⏹️ 主线程监控已停止');
    }
  }

  getReport() {
    const totalTasks = this.longTasks.length;
    const totalBlockTime = this.longTasks.reduce((sum, task) => sum + task.duration, 0);
    const avgBlockTime = totalTasks > 0 ? totalBlockTime / totalTasks : 0;
    const maxBlockTime = Math.max(...this.longTasks.map(task => task.duration), 0);

    return {
      '检测到的长任务数量': totalTasks,
      '总阻塞时间': `${totalBlockTime.toFixed(2)}ms`,
      '平均阻塞时间': `${avgBlockTime.toFixed(2)}ms`,
      '最长阻塞时间': `${maxBlockTime.toFixed(2)}ms`,
      '性能评级': this.getPerformanceGrade(maxBlockTime)
    };
  }

  getPerformanceGrade(maxBlockTime) {
    if (maxBlockTime < 16) return '🟢 优秀 (60FPS)';
    if (maxBlockTime < 33) return '🟡 良好 (30FPS)';
    if (maxBlockTime < 100) return '🟠 一般 (可感知延迟)';
    return '🔴 差 (明显卡顿)';
  }
}

/**
 * 运行时内存性能分析器
 * 结合Performance Timeline API和Memory API进行深度分析
 */
class RuntimeMemoryProfiler {
  constructor() {
    this.memorySnapshots = [];
    this.performanceMarks = [];
    this.isRecording = false;
  }

  startProfiling(label = 'memory-profile') {
    this.isRecording = true;
    this.baseLabel = label;
    
    // 标记开始时间
    performance.mark(`${label}-start`);
    
    // 开始内存采样
    this.memoryInterval = setInterval(() => {
      this.takeMemorySnapshot();
    }, 100); // 每100ms采样一次

    console.log(`🔍 内存性能分析开始: ${label}`);
  }

  takeMemorySnapshot() {
    if (!this.isRecording) return;

    const snapshot = {
      timestamp: performance.now(),
      memory: performance.memory ? {
        used: performance.memory.usedJSHeapSize,
        total: performance.memory.totalJSHeapSize,
        limit: performance.memory.jsHeapSizeLimit
      } : null,
      
      // 获取当前活跃的DOM节点数
      domNodes: document.getElementsByTagName('*').length,
      
      // 获取当前的事件监听器数量（近似）
      eventListeners: this.estimateEventListeners()
    };

    this.memorySnapshots.push(snapshot);
  }

  // 估算事件监听器数量的近似方法
  estimateEventListeners() {
    let count = 0;
    const elements = document.getElementsByTagName('*');
    
    // 这只是一个粗略的估算
    for (let element of elements) {
      // 检查常见的事件属性
      const eventAttrs = ['onclick', 'onload', 'onchange', 'onsubmit'];
      for (let attr of eventAttrs) {
        if (element[attr]) count++;
      }
    }
    
    return count;
  }

  markOperation(operationName) {
    if (!this.isRecording) return;
    
    const markName = `${this.baseLabel}-${operationName}`;
    performance.mark(markName);
    
    this.performanceMarks.push({
      name: operationName,
      timestamp: performance.now(),
      memorySnapshot: this.memorySnapshots[this.memorySnapshots.length - 1]
    });
    
    console.log(`📍 性能标记: ${operationName}`);
  }

  stopProfiling() {
    if (!this.isRecording) return;
    
    this.isRecording = false;
    clearInterval(this.memoryInterval);
    
    // 标记结束时间
    performance.mark(`${this.baseLabel}-end`);
    
    // 计算总体性能指标
    performance.measure(
      `${this.baseLabel}-total`,
      `${this.baseLabel}-start`,
      `${this.baseLabel}-end`
    );

    console.log('⏹️ 内存性能分析结束');
    return this.generateReport();
  }

  generateReport() {
    if (this.memorySnapshots.length === 0) return null;

    const firstSnapshot = this.memorySnapshots[0];
    const lastSnapshot = this.memorySnapshots[this.memorySnapshots.length - 1];
    
    const memoryGrowth = lastSnapshot.memory && firstSnapshot.memory ? 
      lastSnapshot.memory.used - firstSnapshot.memory.used : 0;
    
    const domGrowth = lastSnapshot.domNodes - firstSnapshot.domNodes;
    
    // 计算内存增长率
    const duration = (lastSnapshot.timestamp - firstSnapshot.timestamp) / 1000; // 转换为秒
    const memoryGrowthRate = memoryGrowth / duration; // bytes/second

    // 分析内存趋势
    const trend = this.analyzeMemoryTrend();

    const report = {
      '分析时长': `${duration.toFixed(2)}秒`,
      '内存变化': `${(memoryGrowth / 1024 / 1024).toFixed(2)}MB`,
      '内存增长率': `${(memoryGrowthRate / 1024 / 1024).toFixed(2)}MB/秒`,
      'DOM节点变化': `${domGrowth}个`,
      '内存趋势': trend,
      '性能标记数量': this.performanceMarks.length,
      '采样点数量': this.memorySnapshots.length
    };

    // 输出详细的时间线数据
    console.group('📊 内存性能分析报告');
    console.table(report);
    console.log('📈 内存使用趋势:', this.memorySnapshots.map(s => ({
      时间: `${s.timestamp.toFixed(0)}ms`,
      内存: s.memory ? `${(s.memory.used / 1024 / 1024).toFixed(2)}MB` : 'N/A',
      DOM节点: s.domNodes
    })));
    console.groupEnd();

    return report;
  }

  analyzeMemoryTrend() {
    if (this.memorySnapshots.length < 3) return '数据不足';

    const samples = this.memorySnapshots.slice(-5); // 取最后5个样本
    let increases = 0;
    let decreases = 0;

    for (let i = 1; i < samples.length; i++) {
      if (!samples[i].memory || !samples[i-1].memory) continue;
      
      const change = samples[i].memory.used - samples[i-1].memory.used;
      if (change > 1024 * 1024) increases++; // 增长超过1MB
      else if (change < -1024 * 1024) decreases++; // 减少超过1MB
    }

    if (increases > decreases * 2) return '🔴 持续增长';
    if (decreases > increases * 2) return '🟢 持续下降';
    return '🟡 波动';
  }
}

/**
 * 渲染性能分析器
 * 监控重排重绘、FPS、渲染时间等关键指标
 */
class RenderingPerformanceAnalyzer {
  constructor() {
    this.frameData = [];
    this.paintObserver = null;
    this.isMonitoring = false;
    this.startTime = 0;
  }

  startMonitoring() {
    this.isMonitoring = true;
    this.startTime = performance.now();
    this.frameData = [];

    // 监控渲染事件
    this.setupPaintObserver();
    
    // 监控帧率
    this.startFPSMonitoring();
    
    // 监控布局抖动
    this.startLayoutShiftMonitoring();

    console.log('🎨 渲染性能监控已启动...');
  }

  setupPaintObserver() {
    if ('PerformanceObserver' in window) {
      this.paintObserver = new PerformanceObserver((list) => {
        for (const entry of list.getEntries()) {
          this.recordPaintEvent(entry);
        }
      });

      // 监控多种渲染相关的事件
      this.paintObserver.observe({ 
        entryTypes: ['paint', 'measure', 'navigation', 'layout-shift'] 
      });
    }
  }

  recordPaintEvent(entry) {
    const eventData = {
      type: entry.entryType,
      name: entry.name,
      startTime: entry.startTime,
      duration: entry.duration || 0,
      timestamp: Date.now()
    };

    // 特殊处理layout-shift事件
    if (entry.entryType === 'layout-shift') {
      eventData.value = entry.value;
      eventData.hadRecentInput = entry.hadRecentInput;
      
      if (entry.value > 0.1) {
        console.warn(
          `🚨 布局抖动告警! CLS值: ${entry.value.toFixed(4)}, ` +
          `用户输入相关: ${entry.hadRecentInput}`
        );
      }
    }

    this.frameData.push(eventData);
  }

  startFPSMonitoring() {
    let lastTime = performance.now();
    let frameCount = 0;
    let fps = 0;

    const measureFPS = () => {
      if (!this.isMonitoring) return;

      frameCount++;
      const currentTime = performance.now();
      
      // 每秒计算一次FPS
      if (currentTime - lastTime >= 1000) {
        fps = Math.round((frameCount * 1000) / (currentTime - lastTime));
        
        this.frameData.push({
          type: 'fps',
          name: 'frame-rate',
          startTime: currentTime,
          value: fps,
          timestamp: Date.now()
        });

        // FPS告警
        if (fps < 30) {
          console.warn(`🐌 FPS过低: ${fps}fps, 用户可能感受到卡顿`);
        } else if (fps < 50) {
          console.log(`⚠️ FPS较低: ${fps}fps, 建议优化渲染性能`);
        }

        frameCount = 0;
        lastTime = currentTime;
      }

      requestAnimationFrame(measureFPS);
    };

    requestAnimationFrame(measureFPS);
  }

  startLayoutShiftMonitoring() {
    // 使用ResizeObserver监控元素大小变化
    if ('ResizeObserver' in window) {
      const resizeObserver = new ResizeObserver((entries) => {
        for (const entry of entries) {
          // 记录元素大小变化，可能导致布局抖动
          this.frameData.push({
            type: 'resize',
            name: 'element-resize',
            startTime: performance.now(),
            target: entry.target.tagName,
            newSize: {
              width: entry.contentRect.width,
              height: entry.contentRect.height
            },
            timestamp: Date.now()
          });
        }
      });

      // 监控主要容器元素
      const containers = document.querySelectorAll('main, .container, #app, #root');
      containers.forEach(container => {
        resizeObserver.observe(container);
      });

      this.resizeObserver = resizeObserver;
    }
  }

  stopMonitoring() {
    this.isMonitoring = false;
    
    if (this.paintObserver) {
      this.paintObserver.disconnect();
    }
    
    if (this.resizeObserver) {
      this.resizeObserver.disconnect();
    }

    console.log('⏹️ 渲染性能监控已停止');
    return this.generateRenderingReport();
  }

  generateRenderingReport() {
    const duration = (performance.now() - this.startTime) / 1000;
    
    // 分析各种事件
    const paintEvents = this.frameData.filter(d => d.type === 'paint');
    const layoutShifts = this.frameData.filter(d => d.type === 'layout-shift');
    const fpsData = this.frameData.filter(d => d.type === 'fps');
    const resizeEvents = this.frameData.filter(d => d.type === 'resize');

    // 计算CLS总分
    const totalCLS = layoutShifts.reduce((sum, shift) => sum + (shift.value || 0), 0);
    
    // 计算平均FPS
    const avgFPS = fpsData.length > 0 ? 
      fpsData.reduce((sum, fps) => sum + fps.value, 0) / fpsData.length : 0;

    // 计算最低FPS
    const minFPS = fpsData.length > 0 ? 
      Math.min(...fpsData.map(fps => fps.value)) : 0;

    const report = {
      '监控时长': `${duration.toFixed(2)}秒`,
      '平均FPS': `${avgFPS.toFixed(1)}`,
      '最低FPS': `${minFPS}`,
      '总CLS分数': `${totalCLS.toFixed(4)}`,
      '绘制事件数': paintEvents.length,
      '布局抖动次数': layoutShifts.length,
      '元素调整次数': resizeEvents.length,
      '性能等级': this.calculateRenderingGrade(avgFPS, totalCLS)
    };

    console.group('🎨 渲染性能分析报告');
    console.table(report);
    
    // 显示详细的布局抖动信息
    if (layoutShifts.length > 0) {
      console.log('📊 布局抖动详情:', layoutShifts.map(shift => ({
        时间: `${shift.startTime.toFixed(0)}ms`,

    processedDataItems() {
      return this.allDataItems.map(item => ({
        ...item,
        _styleId: `${item.value}-${item.priority}-${item.category}`, // 样式缓存key
        _formatId: `${item.type}-${item.value}-${item.timestamp}` // 格式化缓存key
      }));
    }
  },
  
  watch: {
    // 优化4：精确的数据监听和缓存更新
    processedDataItems: {
      handler(newItems, oldItems) {
        this.updateCaches(newItems, oldItems);
      },
      deep: false // 浅层监听，避免深度遍历
    },
    
    realTimeData: {
      handler: debounce(function(newData) {
        this.updateChartsSelectively(newData);

      deep: true
    }
  },

    this.startOptimizedPolling();
    
    // 预热缓存
    this.preloadCaches();
  },
  
  methods: {
    // 优化的轮询策略
    startOptimizedPolling() {
      let isVisible = true;
      let pollInterval = 1000; // 动态调整轮询间隔
      
      // 页面可见性检测
      document.addEventListener('visibilitychange', () => {
        isVisible = !document.hidden;
        pollInterval = isVisible ? 1000 : 5000; // 不可见时降低频率
      });
      
      // 智能轮询
      const poll = async () => {
        if (isVisible) {
          try {
            const newData = await this.fetchRealTimeData();
            
            // 只有数据真正变化时才更新
            if (this.hasDataChanged(newData)) {
              this.realTimeData = newData;
            }
          } catch (error) {
            console.error('数据轮询失败:', error);
            pollInterval = Math.min(pollInterval * 2, 10000); // 失败时退避
          }
        }
        
        setTimeout(poll, pollInterval);
      };
      
      poll();
    },
    
    // 缓存管理
    updateCaches(newItems, oldItems) {
      const start = performance.now();
      
      // 增量更新缓存，而不是全量重建
      const newItemsMap = new Map(newItems.map(item => [item.id, item]));
      const oldItemsMap = new Map((oldItems || []).map(item => [item.id, item]));
      
      // 清理已删除项目的缓存
      for (const [id] of oldItemsMap) {
        if (!newItemsMap.has(id)) {
          delete this.cachedRowStyles[id];
          delete this.cachedFormattedData[id];
        }
      }
      
      // 更新变化的项目
      for (const [id, item] of newItemsMap) {
        const oldItem = oldItemsMap.get(id);
        
        // 样式缓存
        if (!oldItem || oldItem._styleId !== item._styleId) {
          this.cachedRowStyles[id] = this.calculateRowStyle(item);
        }
        
        // 格式化数据缓存
        if (!oldItem || oldItem._formatId !== item._formatId) {
          this.cachedFormattedData[id] = this.formatData(item);
        }

    },

      for (const chartId in newData) {
        const chartData = newData[chartId];

          this.cachedChartData[chartId] = this.processChartData(chartData);
          this.chartUpdateTriggers[chartId] = Date.now(); // 触发图表更新
        }
      }
    },
    
    // 预加载缓存
    preloadCaches() {
      // 使用 requestIdleCallback 在浏览器空闲时预加载
      if ('requestIdleCallback' in window) {
        requestIdleCallback(() => {
          this.allDataItems.forEach(item => {
            if (!this.cachedRowStyles[item.id]) {
              this.cachedRowStyles[item.id] = this.calculateRowStyle(item);
            }
            if (!this.cachedFormattedData[item.id]) {
              this.cachedFormattedData[item.id] = this.formatData(item);
            }
          });
        });
      }

    // 优化的样式计算（添加缓存逻辑）
    calculateRowStyle(item) {
      const cacheKey = `${item.value}-${item.priority}-${item.category}`;
      
      if (this.styleCache && this.styleCache[cacheKey]) {
        return this.styleCache[cacheKey];
      }
      
      const style = {
        backgroundColor: this.calculateColor(item.value),
        height: this.calculateHeight(item.priority),
        transform: `translateX(${this.calculateOffset(item.category)}px)`
      };
      
      // 缓存计算结果
      if (!this.styleCache) this.styleCache = {};
      this.styleCache[cacheKey] = style;
      
      return style;
    },
    
    // 数据变化检测

      return JSON.stringify(newData) !== JSON.stringify(this.realTimeData);

        }

      return true;

    this.cachedFormattedData = {};
    this.cachedChartData = {};

    this.monitors.set('blocking', new MainThreadBlockageDetector(30));
    
    // 启动所有监控
    this.monitors.forEach(monitor => monitor.start());
    
    console.log('🛡️ 性能守护者已启动，实时监控应用性能...');
  }
  
  // 生产环境下的轻量级监控
  enableProductionMode() {
    // 只监控关键指标，避免影响性能
    this.setupLightweightMonitoring();
  }
}
