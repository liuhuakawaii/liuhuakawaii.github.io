GSAP 常见case ​

一、ScrollTrigger 核心 API 分类速查表 ​

1️⃣ 触发与区间类 ​

参数类型示例含义
triggerselector/Element".box"触发参考元素
startstring/number"top center"开始滚动点
endstring/number/function"+=500"结束滚动点
endTriggerselector".end"用另一个元素计算 end

start/end 语法

js
"elementPosition viewportPosition"

elementPosition:

css
top center bottom

viewportPosition:

css
top center bottom
百分比 80%

数值：

js
300 = scrollTop 300px
"+=500" = 相对 start

2️⃣ 映射控制 ​

参数类型示例作用
scrubboolean/numbertrue / 1滚动同步
toggleActionsstring"play pause resume reset"非 scrub 模式播放控制

3️⃣ 固定与布局 ​

参数类型示例说明
pinboolean/Elementtrue固定元素
pinSpacingbooleanfalse是否保留占位空间
anticipatePinnumber1提前补偿滚动抖动

4️⃣ 调试与生命周期 ​

参数类型示例用途
markersbooleantrue显示 start/end
onEnterfunction()=>{}进入回调
onLeavefunction()=>{}离开回调
onUpdatefunctionself=>{}滚动更新

gsap.registerPlugin(ScrollTrigger);

/* =====================================================
CASE 1 — 滚动变形动画 (scrub)
用途：官网视觉过渡、Hero 动画
===================================================== */

gsap.to("#shapeBox", {
  scrollTrigger: {
    trigger: "#shapeBox",
    start: "top 80%",
    end: "top 30%",
    scrub: true,
    markers: true
  },
  scale: 1.8,
  rotation: 180,
  borderRadius: "50%"
});


/* =====================================================
CASE 2 — Timeline 分段动画
用途：产品故事演示
===================================================== */

const tl = gsap.timeline({
  scrollTrigger: {
    trigger: "#timelineBox",
    start: "top center",
    end: "bottom center",
    scrub: true
  }
});

tl.to("#timelineBox", {
  x: 400,
  rotation: 90,
  duration: 1
})
.to("#timelineBox", {
  y: -200,
  scale: 0.5,
  backgroundColor: "#22c55e",
  duration: 1
});


/* =====================================================
CASE 3 — Pin 固定区域
用途：苹果官网叙事
===================================================== */

ScrollTrigger.create({
  trigger: "#pinSection",
  start: "top top",
  end: "+=100%",
  pin: true,
  scrub: true,
  anticipatePin: 1
});


/* =====================================================
CASE 4 — 横向滚动
用途：横向画廊
===================================================== */

const horizontal = document.querySelector(".horizontal");

gsap.to(horizontal, {
  x: () => -(horizontal.scrollWidth - window.innerWidth),
  ease: "none",
  scrollTrigger: {
    trigger: ".horizontal-wrapper",
    start: "top top",
    end: () => "+=" + horizontal.scrollWidth,
    pin: true,
    scrub: true
  }
});


/* =====================================================
CASE 5 — 进入播放动画 (非 scrub)
用途：进入视口播放一次
===================================================== */

gsap.from("#fadeBox", {
  scrollTrigger: {
    trigger: "#fadeBox",
    start: "top 80%",
    toggleActions: "play none none reverse"
  },
  opacity: 0,
  y: 100,
  duration: 1
});


/* =====================================================
CASE 6 — 滚动进度监听
用途：进度条 / 数字反馈
===================================================== */

ScrollTrigger.create({
  trigger: document.body,
  start: "top top",
  end: "bottom bottom",
  onUpdate(self) {
    document.getElementById("progressText").innerText =
      Math.round(self.progress * 100) + "%";
  }
});

</script>

</body>
