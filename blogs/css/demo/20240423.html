  const Temperature = ({ type, temperatureValue, setTemperatureValue, keepSeed, setKeepSeed }) => {
    const sliderRef = useRef(null);
    const step = 0.5;
    const max = 10;
    const min = 0;
    const numOfSteps = (max - min) / step;

    // const handleChange = (event) => {
    //     setValue(event.target.value);
    // };


    const handleCheckBox = (e) => {
        e.stopPropagation()
        setKeepSeed(pre => !pre)
    }

    const updateValue = (newVal) => {
        if (newVal >= min && newVal <= max) {
            setTemperatureValue(newVal);
        }
    };

    const handleClick = (event) => {
        event.stopPropagation()
        const slider = sliderRef.current;
        const rect = slider.getBoundingClientRect();
        const clickedValue = ((event.clientX - rect.left) / rect.width) * max;
        updateValue(Math.round(clickedValue / step) * step);
    };

    const handleMove = (clientX) => {
        const slider = sliderRef.current;
        const rect = slider.getBoundingClientRect();
        const movedValue = ((clientX - rect.left) / rect.width) * max;
        updateValue(Math.round(movedValue / step) * step);
    };

    const handleMouseDown = (event) => {
        event.stopPropagation()
        handleMove(event.clientX);
        document.addEventListener('mousemove', handleMouseMove);
        document.addEventListener('mouseup', handleMouseUp);
    };

    const handleMouseMove = (event) => {
        handleMove(event.clientX);
    };

    const handleMouseUp = () => {
        document.removeEventListener('mousemove', handleMouseMove);
        document.removeEventListener('mouseup', handleMouseUp);
    };

    const handleTouchStart = (event) => {
        event.stopPropagation()
        handleMove(event.touches[0].clientX);
    };

    const handleTouchMove = (event) => {
        handleMove(event.touches[0].clientX);
    };

    const positionStyle = type === 'redo'
        ? { left: 0, transform: `translateY(calc(-100% - 10px))` }
        : { left: '50%', transform: ` translate(-50%, calc(-100% - 10px))` }

    return (
        <div className={style.temperatureWrapper} style={ ...positionStyle }>
            <div className={style.temperatureSlider}>
                <div className={style.temperatureHeader}>
                    <span>Temperature</span>
                    {type === 'redo' && <label className={style.checkBoxContainer} onClick={handleCheckBox}>
                        <span className={`${style.checkmark} ${keepSeed ? style.checked : ""}`}></span>
                        <span>Keep Seed</span>
                    </label>}
                </div>
                <div className={style.sliderContainer} onClick={handleClick}>
                    <div className={style.slidertrack} ref={sliderRef}>
                        <div className={style.sliderInner}>
                            {[...Array(numOfSteps + 1)].map((_, index) => (
                                <div
                                    style={
                                        left: `${index * 5}%`,
                                        display: `${[0, 20].includes(index) ? 'none' : 'block'}`
                                    }
                                    key={index}
                                    className={style.sliderstep}></div>
                            ))}
                            <div
                                className={style.sliderthumb}
                                style={ left: `${(temperatureValue / max) * 100}%` }
                                onMouseDown={handleMouseDown}
                                onTouchStart={handleTouchStart}
                                onTouchMove={handleTouchMove}
                            ></div>
                            <div className={style.rangeBg} style={
                                width: `${(temperatureValue / max) * 100}%`,
                            }></div>
                        </div>
                    </div>
                    <div className={style.rangeVal}>{temperatureValue}</div>
                </div>
            </div>
            <div className={style.bottomHodler}></div>
        </div>
    );
};

const RangeSlider = ({
  initialValue = 0,
  min = 0,
  max = 100
}) => {
  const maxRange = max - min
  const [positionValue, setPositionValue] = useState(0);
  const [showValue, setShowValue] = useState(0);
  const rangeRef = useRef(null);
  const knobRef = useRef(null);

  const handleMouseMove = (e) => {
    if (e.button !== 0) return
    const startX = e.clientX;
    const selfXPosition = knobRef.current.offsetLeft
    document.onmousemove = (e) => {
      const endX = e.clientX;
      let diffX = selfXPosition + endX - startX
      if (diffX <= 0) {
        diffX = 0
      }
      if (diffX >= (rangeRef.current.clientWidth - knobRef.current.clientWidth)) {
        diffX = (rangeRef.current.clientWidth - knobRef.current.clientWidth)
      }
      setPositionValue(diffX);
    }
    document.onmouseup = function () {
      document.onmousemove = null;
      document.onmouseup = null;
    };
  }

  useEffect(() => {
    if (initialValue) {
      setShowValue(Math.round((initialValue / (rangeRef.current.clientWidth - knobRef.current.clientWidth)) * maxRange))
    }
  }, [initialValue])


  useEffect(() => {
    setShowValue(Math.round((positionValue / (rangeRef.current.clientWidth - knobRef.current.clientWidth)) * maxRange))
  }, [positionValue])


  return (
    <div className={style.rangeWrapper}>
      <div className={style.rangeImg}>
        <img src={range_controler} alt="range_controler" />
      </div>
      <div className={style.rangeControler}>
        <div className={style.rangeTool}>
          <div className={style.title}>强度</div>
          <div className={style.toolMain}>
            <span className={style.value}>{showValue}</span>
            <img className={style.settings} src={pinch_settings} alt="pinch_settings" />
            <img className={style.follow} src={pinch_follow} alt="pinch_follow" />
          </div>
        </div>
        <div className={style.rangeSlider} ref={rangeRef}>
          <div
            className={style.sliderKnob}
            onMouseDown={handleMouseMove}
            style={ left: `${positionValue}px` }
            ref={knobRef}></div>
        </div>
      </div>

    </div >
  );

}
