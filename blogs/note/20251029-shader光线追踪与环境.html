🌅 Shader 进阶 - 光线追踪与环境映射的极致渲染 ​

🌟 传统渲染 vs 光线追踪 ​

在前面所有文章中,我们使用的是光栅化渲染 (Rasterization):

光栅化流程:
顶点 → 三角形 → 光栅化 → 像素

但还有另一种完全不同的思路 - 光线追踪 (Ray Tracing):

光线追踪流程:
相机 → 发射光线 → 碰撞检测 → 着色
特性光栅化光线追踪
原理投影三角形到屏幕模拟真实光路
性能⚡⚡⚡ 极快🐢 慢 (除非硬件RT)
反射/折射❌ 需要技巧 (环境贴图)✅ 天然支持
软阴影❌ 困难✅ 简单
全局光照❌ 需要烘焙✅ 自然产生
复杂几何✅ 三角形数量决定✅ SDF可表示任意形状

GPU上的挑战: 传统光线追踪需要加速结构 (BVH),但在Shader中不易实现。

解决方案: Ray Marching - 用 SDF 替代三角形!

🚶 Ray Marching - GPU上的光线追踪 ​

🎯 核心思想 ​

不直接求光线与物体的交点,而是沿着光线逐步行进:

     摄像机               SDF球体
       ●                    ●
       │╲
       │ ╲ 光线
       │  ╲
       │   ●───→ 步进1 (安全距离d1)
       │        ╲
       │         ●──→ 步进2 (安全距离d2)
       │            ╲
       │             ●─→ 步进3 (d3≈0,命中!)

算法步骤:

  1. 从摄像机发射光线
  2. 查询当前点到最近表面的距离 d = SDF(点)
  3. 沿光线前进 d (确保不会穿过物体)
  4. 重复直到 d < 阈值 (命中) 或超出范围

📐 数学表达 ​

光线方程: P(t) = origin + t × direction

伪代码:
t = 0
for i = 0 to MAX_STEPS:
    point = origin + t * direction
    dist = SDF(point)
    
    if dist < EPSILON:
        return HIT
    
    t += dist  // 安全步进
    
    if t > MAX_DISTANCE:
        return MISS

🔧 基础实现 ​

glsl
#ifdef GL_ES
precision highp float;
#endif

uniform vec2 u_resolution;
uniform float u_time;

const int MAX_STEPS = 100;
const float MAX_DIST = 100.0;
const float EPSILON = 0.001;

// 🔵 SDF: 球体
float sdSphere(vec3 p, float radius) {
    return length(p) - radius;
}

// 🔷 SDF: 盒子
float sdBox(vec3 p, vec3 b) {
    vec3 q = abs(p) - b;
    return length(max(q, 0.0)) + min(max(q.x, max(q.y, q.z)), 0.0);
}

// 🌍 场景SDF (所有物体的组合)
float sceneSDF(vec3 p) {
    // 🔵 一个球 (中心在原点,半径1.0)
    float sphere = sdSphere(p - vec3(0.0, 1.0, 0.0), 1.0);
    
    // 📦 地面 (无限平面)
    float ground = p.y;
    
    // 🎭 组合: 取最小值 = 并集
    return min(sphere, ground);
}

// 🚶 Ray Marching 主函数
float rayMarch(vec3 ro, vec3 rd) {
    float t = 0.0;  // 当前距离
    
    for(int i = 0; i < MAX_STEPS; i++) {
        vec3 p = ro + t * rd;  // 当前点
        float dist = sceneSDF(p);
        
        if(dist < EPSILON) {
            return t;  // 命中!
        }
        
        t += dist;  // 安全步进
        
        if(t > MAX_DIST) {
            break;  // 超出范围
        }
    }
    
    return -1.0;  // 未命中
}

void main() {
    // 📐 归一化设备坐标 (NDC)
    vec2 uv = (gl_FragCoord.xy - 0.5 * u_resolution.xy) / u_resolution.y;
    
    // 📹 摄像机设置
    vec3 ro = vec3(0.0, 1.0, -3.0);  // 相机位置 (Ray Origin)
    vec3 rd = normalize(vec3(uv, 1.0));  // 光线方向 (Ray Direction)
    
    // 🚶 执行 Ray Marching
    float t = rayMarch(ro, rd);
    
    // 🎨 着色
    vec3 color = vec3(0.0);
    
    if(t > 0.0) {
        // 命中物体
        color = vec3(0.5, 0.7, 1.0);  // 蓝色
    } else {
        // 背景
        color = vec3(0.1, 0.1, 0.2);  // 深蓝
    }
    
    gl_FragColor = vec4(color, 1.0);
}

效果: 渲染出一个蓝色球体和地面!

💡 Ray Marching 光照 ​

🧭 计算法线 ​

SDF的强大之处:法线可以通过梯度计算!

法线 = ∇SDF = (∂SDF/∂x, ∂SDF/∂y, ∂SDF/∂z)

数值近似 (中心差分):
N.x ≈ SDF(p + εx) - SDF(p - εx)
N.y ≈ SDF(p + εy) - SDF(p - εy)
N.z ≈ SDF(p + εz) - SDF(p - εz)

🔧 实现 ​

glsl
// 🧭 计算法线
vec3 getNormal(vec3 p) {
    float eps = 0.001;  // 微小偏移
    vec2 h = vec2(eps, 0.0);
    
    return normalize(vec3(
        sceneSDF(p + h.xyy) - sceneSDF(p - h.xyy),
        sceneSDF(p + h.yxy) - sceneSDF(p - h.yxy),
        sceneSDF(p + h.yyx) - sceneSDF(p - h.yyx)
    ));
}

☀️ 添加光照 ​

glsl
uniform vec3 u_lightPos;

void main() {
    vec2 uv = (gl_FragCoord.xy - 0.5 * u_resolution.xy) / u_resolution.y;
    
    vec3 ro = vec3(0.0, 1.0, -3.0);
    vec3 rd = normalize(vec3(uv, 1.0));
    
    float t = rayMarch(ro, rd);
    
    vec3 color = vec3(0.0);
    
    if(t > 0.0) {
        // 📍 命中点
        vec3 p = ro + t * rd;
        
        // 🧭 法线
        vec3 N = getNormal(p);
        
        // 💡 光照方向
        vec3 L = normalize(u_lightPos - p);
        
        // ☀️ Lambert 漫反射
        float diff = max(dot(N, L), 0.0);
        
        // 🎨 基础颜色
        vec3 baseColor = vec3(0.5, 0.7, 1.0);
        
        // 🌍 环境光
        float ambient = 0.2;
        
        color = baseColor * (ambient + diff * 0.8);
    } else {
        // 🌌 背景
        color = vec3(0.1, 0.1, 0.2);
    }
    
    gl_FragColor = vec4(color, 1.0);
}

🌑 软阴影 ​

Ray Marching 可以轻松实现软阴影 (Soft Shadows):

glsl
// 🌑 软阴影计算
float softShadow(vec3 ro, vec3 rd, float mint, float maxt, float k) {
    float res = 1.0;  // 初始为完全无阴影
    float t = mint;
    
    for(int i = 0; i < 50; i++) {
        float h = sceneSDF(ro + rd * t);
        
        if(h < 0.001) {
            return 0.0;  // 完全遮蔽
        }
        
        // 🌫️ 半影计算 (越接近物体,阴影越重)
        res = min(res, k * h / t);
        
        t += h;
        
        if(t > maxt) break;
    }
    
    return res;
}

void main() {
    // ... (之前的代码)
    
    if(t > 0.0) {
        vec3 p = ro + t * rd;
        vec3 N = getNormal(p);
        vec3 L = normalize(u_lightPos - p);
        
        // 💡 漫反射
        float diff = max(dot(N, L), 0.0);
        
        // 🌑 计算阴影 (从命中点向光源发射光线)
        float shadow = softShadow(p + N * 0.01, L, 0.01, 5.0, 8.0);
        //                        ↑             ↑    ↑     ↑   ↑
        //                      起点偏移      光线方向  起始  最大  锐度
        
        diff *= shadow;  // 应用阴影
        
        vec3 baseColor = vec3(0.5, 0.7, 1.0);
        float ambient = 0.2;
        
        color = baseColor * (ambient + diff * 0.8);
    }
    
    gl_FragColor = vec4(color, 1.0);
}

参数说明:

  • k (锐度): 越大阴影边缘越锐利 (8.0 ~ 128.0)
  • mint, maxt: 阴影检测范围

🔮 环境贴图 (Environment Mapping) ​

🎯 什么是环境贴图? ​

用2D图像存储周围环境的光照信息:

真实场景 (360°全景) → 映射到 2D 纹理 → 用于反射/折射

两种主要格式:

  1. 球形贴图 (Spherical / Equirectangular)
  2. 立方体贴图 (Cubemap)

🌐 球形环境贴图 ​

📐 原理 ​

将 3D 方向向量映射到 2D UV:

3D方向 (x, y, z) → (θ, φ) → UV (u, v)

θ = atan(z, x)         // 方位角 (-π ~ π)
φ = asin(y)            // 仰角 (-π/2 ~ π/2)

u = (θ + π) / (2π)     // 映射到 [0, 1]
v = (φ + π/2) / π      // 映射到 [0, 1]

可视化:

      北极 (0, 1, 0)
         ↑
         │
    ──────●────── 赤道面
        ╱│╲
       ╱ │ ╲
      ↙  ↓  ↘
   南极 (0, -1, 0)

展开成矩形:
┌─────────────────┐ ← 北极 (v=1)
│   全景图像      │
│                 │
│                 │
└─────────────────┘ ← 南极 (v=0)

🔧 实现 ​

glsl
uniform sampler2D u_envMap;  // 球形环境贴图

const float PI = 3.14159265359;

// 🌐 将3D方向转换为球形UV
vec2 directionToSphericalUV(vec3 dir) {
    // 📐 计算球坐标
    float theta = atan(dir.z, dir.x);  // 水平角 [-π, π]
    float phi = asin(dir.y);           // 垂直角 [-π/2, π/2]
    
    // 🗺️ 映射到 [0, 1]
    vec2 uv;
    uv.x = (theta + PI) / (2.0 * PI);
    uv.y = (phi + PI * 0.5) / PI;
    
    return uv;
}

// 🎨 采样环境贴图
vec3 sampleEnvironment(vec3 direction) {
    vec2 uv = directionToSphericalUV(normalize(direction));
    return texture2D(u_envMap, uv).rgb;
}

void main() {
    vec2 uv = (gl_FragCoord.xy - 0.5 * u_resolution.xy) / u_resolution.y;
    
    vec3 ro = vec3(0.0, 1.0, -3.0);
    vec3 rd = normalize(vec3(uv, 1.0));
    
    float t = rayMarch(ro, rd);
    
    vec3 color;
    
    if(t > 0.0) {
        // 命中物体 (后续添加反射)
        color = vec3(0.5);
    } else {
        // 🌌 背景 = 环境贴图
        color = sampleEnvironment(rd);
    }
    
    gl_FragColor = vec4(color, 1.0);
}

🎲 立方体贴图 (Cubemap) ​

📐 原理 ​

用6个面的正方形纹理组成一个立方体:

        ┌───┐
        │ Y+│ (顶部)
    ┌───┼───┼───┬───┐
    │X- │Z+ │X+ │Z- │
    └───┼───┼───┴───┘
        │ Y-│ (底部)
        └───┘

采样时:根据方向向量,选择对应的面

🔧 Three.js中使用 Cubemap ​

javascript
// JavaScript 加载 Cubemap

]);

const material = new THREE.ShaderMaterial({
    uniforms: {
        u_envMap: { value: envMap }
    },
    fragmentShader: shaderCode
});

GLSL中采样:

glsl
uniform samplerCube u_envMap;  // Cubemap类型

void main() {
    vec3 direction = ...;  // 采样方向
    
    // 📦 直接用3D方向采样 (GPU自动选择面)
    vec3 envColor = textureCube(u_envMap, direction).rgb;
    
    gl_FragColor = vec4(envColor, 1.0);
}

优势:

  • ✅ 更少的失真 (相比球形贴图)
  • ✅ GPU硬件优化
  • ✅ 边缘过滤更好

🪞 反射 (Reflection) ​

📐 原理 ​

根据法线和视线方向计算反射光线:

      N (法线)
      ↑
      │
  V ╱   ╲ R (反射)
   ╱     ╲
  ╱       ↘

R = reflect(V, N) = V - 2(V·N)N

🔧 镜面反射实现 ​

glsl
uniform samplerCube u_envMap;

void main() {
    vec2 uv = (gl_FragCoord.xy - 0.5 * u_resolution.xy) / u_resolution.y;
    
    vec3 ro = vec3(0.0, 1.0, -3.0);
    vec3 rd = normalize(vec3(uv, 1.0));
    
    float t = rayMarch(ro, rd);
    
    vec3 color;
    
    if(t > 0.0) {
        vec3 p = ro + t * rd;
        vec3 N = getNormal(p);
        
        // 🪞 计算反射方向
        vec3 V = -rd;  // 视线方向(指向摄像机)
        vec3 R = reflect(-V, N);  // 反射方向
        
        // 🎨 采样环境贴图
        color = textureCube(u_envMap, R).rgb;
        
        // 💡 可选:添加菲涅尔效果
        float fresnel = pow(1.0 - max(dot(V, N), 0.0), 5.0);
        vec3 baseColor = vec3(0.2, 0.3, 0.5);
        color = mix(baseColor, color, fresnel);
        
    } else {
        color = textureCube(u_envMap, rd).rgb;
    }
    
    gl_FragColor = vec4(color, 1.0);
}

💎 折射 (Refraction) ​

📐 斯涅尔定律 (Snell's Law) ​

n₁ × sin(θ₁) = n₂ × sin(θ₂)

其中:
- n₁: 入射介质折射率 (空气 ≈ 1.0)
- n₂: 折射介质折射率 (玻璃 ≈ 1.5, 水 ≈ 1.33)
- θ₁, θ₂: 入射角、折射角

可视化:

  空气 (n=1.0)
────────────── 界面
  玻璃 (n=1.5)

    ↓ 入射光
    │ θ₁
────●────
    │╲ θ₂ (变小!)
    │ ╲
    ↓  ↘ 折射光

🔧 实现折射 ​

glsl
uniform samplerCube u_envMap;
uniform float u_ior;  // Index of Refraction (折射率)

void main() {
    vec2 uv = (gl_FragCoord.xy - 0.5 * u_resolution.xy) / u_resolution.y;
    
    vec3 ro = vec3(0.0, 1.0, -3.0);
    vec3 rd = normalize(vec3(uv, 1.0));
    
    float t = rayMarch(ro, rd);
    
    vec3 color;
    
    if(t > 0.0) {
        vec3 p = ro + t * rd;
        vec3 N = getNormal(p);
        vec3 V = -rd;
        
        // 💎 计算折射方向
        float eta = 1.0 / u_ior;  // 空气→玻璃,使用倒数
        vec3 T = refract(-V, N, eta);  // GLSL内置函数
        
        // 🎨 采样环境贴图
        color = textureCube(u_envMap, T).rgb;
        
        // ✨ 组合反射和折射 (Fresnel混合)
        vec3 R = reflect(-V, N);
        vec3 reflectColor = textureCube(u_envMap, R).rgb;
        vec3 refractColor = textureCube(u_envMap, T).rgb;
        
        float fresnel = pow(1.0 - max(dot(V, N), 0.0), 5.0);
        color = mix(refractColor, reflectColor, fresnel);
        
    } else {
        color = textureCube(u_envMap, rd).rgb;
    }
    
    gl_FragColor = vec4(color, 1.0);
}

🌈 色散 (Chromatic Dispersion) ​

不同波长的光折射率不同:

glsl
// 🌈 模拟棱镜效果
void main() {
    // ... (之前的代码)
    
    if(t > 0.0) {
        vec3 p = ro + t * rd;
        vec3 N = getNormal(p);
        vec3 V = -rd;
        
        // 📊 RGB三个通道用不同折射率
        float etaR = 1.0 / 1.50;  // 红光
        float etaG = 1.0 / 1.52;  // 绿光
        float etaB = 1.0 / 1.54;  // 蓝光
        
        vec3 TR = refract(-V, N, etaR);
        vec3 TG = refract(-V, N, etaG);
        vec3 TB = refract(-V, N, etaB);
        
        float r = textureCube(u_envMap, TR).r;
        float g = textureCube(u_envMap, TG).g;
        float b = textureCube(u_envMap, TB).b;
        
        color = vec3(r, g, b);
    }
    
    gl_FragColor = vec4(color, 1.0);
}

效果: 玻璃边缘出现彩虹色!

🌅 IBL (Image-Based Lighting) - 基于图像的光照 ​

🎯 原理 ​

不使用点光源,而是整个环境贴图都是光源!

步骤:

  1. 漫反射IBL: 对环境贴图进行卷积,预计算辐照度
  2. 镜面反射IBL: 基于粗糙度预过滤环境贴图

🔧 简化漫反射IBL ​

glsl
uniform samplerCube u_envMap;
uniform samplerCube u_irradianceMap;  // 预计算的辐照度贴图

void main() {
    vec2 uv = (gl_FragCoord.xy - 0.5 * u_resolution.xy) / u_resolution.y;
    
    vec3 ro = vec3(0.0, 1.0, -3.0);
    vec3 rd = normalize(vec3(uv, 1.0));
    
    float t = rayMarch(ro, rd);
    
    vec3 color;
    
    if(t > 0.0) {
        vec3 p = ro + t * rd;
        vec3 N = getNormal(p);
        vec3 V = -rd;
        
        // 🌍 漫反射IBL (用法线方向采样)
        vec3 irradiance = textureCube(u_irradianceMap, N).rgb;
        vec3 diffuse = irradiance * vec3(0.5, 0.7, 1.0);  // 基础颜色
        
        // ✨ 镜面反射IBL (用反射方向采样)
        vec3 R = reflect(-V, N);
        vec3 specular = textureCube(u_envMap, R).rgb;
        
        // 🎨 Fresnel混合
        float fresnel = pow(1.0 - max(dot(V, N), 0.0), 5.0);
        
        color = mix(diffuse, specular, fresnel);
    } else {
        color = textureCube(u_envMap, rd).rgb;
    }
    
    gl_FragColor = vec4(color, 1.0);
}

🎨 综合实战: 玻璃球 Ray Marching ​

完整场景 ​

glsl
#ifdef GL_ES
precision highp float;
#endif

uniform vec2 u_resolution;
uniform float u_time;
uniform samplerCube u_envMap;

const int MAX_STEPS = 100;
const float MAX_DIST = 100.0;
const float EPSILON = 0.001;
const float PI = 3.14159265359;

// ============ SDF 函数 ============

float sdSphere(vec3 p, float r) {
    return length(p) - r;
}

float sdBox(vec3 p, vec3 b) {
    vec3 q = abs(p) - b;
    return length(max(q, 0.0)) + min(max(q.x, max(q.y, q.z)), 0.0);
}

// ============ 场景组合 ============

float sceneSDF(vec3 p) {
    // 🔮 玻璃球 (ID=1)
    float sphere = sdSphere(p - vec3(0.0, 1.0, 0.0), 1.0);
    
    // 📦 地面 (ID=2)
    float ground = p.y;
    
    return min(sphere, ground);
}

// 获取材质ID
float getMaterialID(vec3 p) {
    float sphere = sdSphere(p - vec3(0.0, 1.0, 0.0), 1.0);
    float ground = p.y;
    
    return (sphere < ground) ? 1.0 : 2.0;  // 1=玻璃, 2=地面
}

// ============ Ray Marching ============

float rayMarch(vec3 ro, vec3 rd) {
    float t = 0.0;
    for(int i = 0; i < MAX_STEPS; i++) {
        vec3 p = ro + t * rd;
        float dist = sceneSDF(p);
        
        if(dist < EPSILON) return t;
        
        t += dist;
        
        if(t > MAX_DIST) break;
    }
    return -1.0;
}

// ============ 法线计算 ============

vec3 getNormal(vec3 p) {
    vec2 h = vec2(0.001, 0.0);
    return normalize(vec3(
        sceneSDF(p + h.xyy) - sceneSDF(p - h.xyy),
        sceneSDF(p + h.yxy) - sceneSDF(p - h.yxy),
        sceneSDF(p + h.yyx) - sceneSDF(p - h.yyx)
    ));
}

// ============ 主渲染 ============

void main() {
    vec2 uv = (gl_FragCoord.xy - 0.5 * u_resolution.xy) / u_resolution.y;
    
    // 📹 摄像机旋转
    float angle = u_time * 0.2;
    vec3 ro = vec3(sin(angle) * 4.0, 2.0, cos(angle) * 4.0);
    vec3 target = vec3(0.0, 1.0, 0.0);
    
    vec3 forward = normalize(target - ro);
    vec3 right = normalize(cross(vec3(0.0, 1.0, 0.0), forward));
    vec3 up = cross(forward, right);
    
    vec3 rd = normalize(uv.x * right + uv.y * up + 1.5 * forward);
    
    // 🚶 Ray Marching
    float t = rayMarch(ro, rd);
    
    vec3 color;
    
    if(t > 0.0) {
        vec3 p = ro + t * rd;
        vec3 N = getNormal(p);
        vec3 V = -rd;
        
        float matID = getMaterialID(p);
        
        if(matID < 1.5) {
            // 🔮 玻璃材质
            vec3 R = reflect(-V, N);
            vec3 T = refract(-V, N, 1.0 / 1.5);
            
            vec3 reflectColor = textureCube(u_envMap, R).rgb;
            vec3 refractColor = textureCube(u_envMap, T).rgb;
            
            // Fresnel
            float fresnel = pow(1.0 - max(dot(V, N), 0.0), 5.0);
            fresnel = fresnel * 0.9 + 0.1;  // 确保最少10%反射
            
            color = mix(refractColor, reflectColor, fresnel);
            
        } else {
            // 🏗️ 地面材质 (简单漫反射)
            vec3 L = normalize(vec3(1.0, 1.0, -0.5));
            float diff = max(dot(N, L), 0.0);
            
            vec3 baseColor = vec3(0.3, 0.3, 0.4);
            color = baseColor * (0.3 + diff * 0.7);
        }
    } else {
        // 🌌 背景
        color = textureCube(u_envMap, rd).rgb;
    }
    
    // 📸 色调映射
    color = color / (color + vec3(1.0));
    color = pow(color, vec3(1.0 / 2.2));  // Gamma校正
    
    gl_FragColor = vec4(color, 1.0);
}
