    // ğŸ“– è¯»å–å½“å‰æµ“åº¦ (r = A, g = B)

    float B = current.g;

    laplacian += texture2D(u_buffer, uv + vec2(0.0, -texel.y)).rg * 0.2;
    laplacian += texture2D(u_buffer, uv + vec2(0.0,  texel.y)).rg * 0.2;

    float laplaceB = laplacian.g;

    B += dB * u_dt;

    A = clamp(A, 0.0, 1.0);
    B = clamp(B, 0.0, 1.0);
    
    gl_FragColor = vec4(A, B, 0.0, 1.0);
}
