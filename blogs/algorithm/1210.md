##  数据结构---队列     
1. 队列结构的封装         
  + 基于数组形式封装      
        ```js   
            class Queue {
              items = [] //前面加#号就不能被访问了，私有属性
              dequeue() { //出队列
                return this.items.shift() //如果有数组模拟队列
                //在shift的过程中，后续索引都会往前移动一位，如果队列过长性能将会不好
              }
              enqueue(data) { //入队列
                this.items.push(data)
              }
              front() { //获取队头元素
                // return this.items[this.items.length - 1]
                return this.items.at(0)
              }
              isEmpty() { //队列判空
                return this.items.length === 0
              }
              size() {
                return this.items.length
              }
              clear() {
                this.items = []
              }
            }     
        ```     
  + 基于对象形式封装      
      ```js   
           class Queue {
              #items = {} //前面加#号就不能被访问了，私有属性
              #lowCount = 0
              #count = 0
              dequeue() { //出队列
                if(this.isEmpty()) return undefined
                let res = this.#items[this.#lowCount]
                delete this.#items[this.#lowCount]
                this.#lowCount++
                return res 
              }
              enqueue(data) { //入队列
                this.#items[this.#count] = data 
                this.#count++
              }
              front() { //获取队头元素
                // return this.items[this.items.length - 1]
                return this.#items[this.#lowCount]
              }
              isEmpty() { //队列判空
                return this.size() === 0
              }
              size() {
                return this.#count - this.#lowCount 
              }
              clear() {
                this.#items = {}
                this.#count = 0
                this.#lowCount = 0
              }
            }
            let queue = new Queue()   
      ```       

2. 击鼓传花游戏     
    ```js   
            function jiguGame(list,num){
              let queue =new Queue()
                for(let i=0;i<list.length;i++){
                  queue.enqueue(list[i])
                }
                while(queue.size()>1){
                  for(let i = 0;i<num;i++){
                    queue.enqueue(queue.dequeue())
                  }
                  queue.dequeue()
                }
                return queue.front()
            }     
    ```     

## 双端队列         
1. 代码     
   ```js   
              class DeQueue {  //dounble-end queue
                #items = {} //前面加#号就不能被访问了，私有属性
                #lowCount = 0
                #count = 0
                removeFront() { //出队列
                  if(this.isEmpty()) return undefined
                  let res = this.#items[this.#lowCount]
                  delete this.#items[this.#lowCount]
                  this.#lowCount++
                  return res 
                }
                removeBack(){
                  if(this.isEmpty()){
                    return
                  }
                  this.#count--
                  const res = this.#items[this.#count]
                  delete this.#items[this.#count]
                  return res
                }
                addBack(data) { //后入队列
                  this.#items[this.#count] = data 
                  this.#count++
                }
                addFront(data){//前入队列
                  //1.如果为空
                  if(this.isEmpty()){
                    this.addBack()
                  }else{
                    //2.lowCount>0
                    if(this.#lowCount>0){
                      this.#lowCount--
                      this.#items[this.#lowCount] = data
                    }else{
                      //lowCount==0
                      for(let i = this.#count;i>0;i--){
                        this.#items[i] = this.#items[i-1]
                      }
                      this.#items[0] = data
                      this.#count++
                    }
                  }
                }
                peekFront() { //获取队头元素
                  // return this.items[this.items.length - 1]
                  return this.#items[this.#lowCount]
                }
                peekBack(){//获取队尾元素
                  if(this.isEmpty()) return
                  return this.#items[this.#count - 1]
                }
                isEmpty() { //队列判空
                  return this.size() === 0
                }
                size() {
                  return this.#count - this.#lowCount 
                }
                clear() {
                  this.#items = {}
                  this.#count = 0
                  this.#lowCount = 0
                }
              }     
    ```     

2. 回文检查     
    ```js     
            //对于不区分大小写的有空格的回文认定  '  da  D'认为是回文
            function test(str){
              //拆分数组再合并成字符串，
              //toLocaleLowerCase()和toLowerCase()的区别是前者会注重地区转换的差异
              const lowStr = str.toLocaleLowerCase().split(' ').join('')
              let dequeue = new DeQueue()
              for(let i = 0;i<lowStr.length;i++){
                dequeue.addBack(lowStr[i])
              }
              let isEqual = true
              while(dequeue.size() > 1){
                if(dequeue.removeFront()!==dequeue.removeBack()){
                  isEqual = false
                  break
                }
              }
              return isEqual
            }
    ```
